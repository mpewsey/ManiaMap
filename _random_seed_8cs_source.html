112 {
113 Seed = seed;
114 Position1 = 0;
115 Position2 = 21;
116 var mj = seed == int.MinValue ? int.MaxValue : Math.Abs(seed);
117 mj = 161803398 - mj;
118 Seeds[55] = mj;
119 var mk = 1;
120
121 // Populate random seed array.
122 for (int i = 1; i < 55; i++)
123 {
124 var index = (21 * i) % 55;
125 Seeds[index] = mk;
126 mk = Mod(mj - mk);
127 mj = Seeds[index];
