133 private void AddCollectable(string group, int id)
134 {
135 // Draw a collectable spot and assign the collectable to the room.
136 var weights = GetWeights(group);
137 var index = RandomSeed.DrawWeightedIndex(weights);
138
139 if (index < 0)
140 throw new Exception($"Failed to find collectable spot for group: {group}.");
141
142 var spot = CollectableSpots[index];
143 CollectableSpots.RemoveAt(index);
144 var room = Layout.Rooms[spot.Room];
145 room.Collectables.Add(spot.Position, id);
146
147 // Update the draw weights of neighboring spots
148 UpdateNeighborWeights(spot);
149 }

156 private void UpdateNeighborWeights(CollectableSpot spot)
157 {
158 var cluster = Clusters[spot.Room];
159
160 foreach (var other in CollectableSpots)
161 {
162 if (!cluster.Contains(other.Room))
163 continue;
164
165 if (other.Room == spot.Room)
166 {
167 // Spots are in the same room. Use distance between collectables.
168 var room = Layout.Rooms[spot.Room];
169 var distance = GetDistance(room, spot.Position, other.Position);
170 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
171 }
172 else
173 {
174 // Spots are in different rooms. Use total distance to connecting doors.
175 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
176 var fromRoom = Layout.Rooms[connection.FromRoom];
177 var toRoom = Layout.Rooms[connection.ToRoom];
178 var fromPosition = spot.Position;
179 var toPosition = other.Position;
180
181 if (connection.ToRoom == spot.Room)
182 (fromPosition, toPosition) = (toPosition, fromPosition);
183
184 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
185 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
186 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
187 }
188 }

283 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
284 {
285 if (!Distances.TryGetValue(room.Template, out var positions))
286 {
287 positions = new Dictionary<Vector2DInt, Array2D<int>>();
288 Distances.Add(room.Template, positions);
289 }
290
291 if (positions.TryGetValue(index1, out var distances))
292 return distances[index2.X, index2.Y];
293 if (positions.TryGetValue(index2, out distances))
294 return distances[index1.X, index1.Y];
295
296 distances = room.Template.FindCellDistances(index1);
297 positions.Add(index1, distances);
298 return distances[index2.X, index2.Y];
299 }
