158 private void AddCollectable(string group, int id)
159 {
160 // Draw a collectable spot and assign the collectable to the room.
161 var weights = GetWeights(group);
162 var index = RandomSeed.DrawWeightedIndex(weights);
163
164 if (index < 0)
165 throw new CollectableSpotNotFoundException($"Failed to find collectable spot for group: {group}.");
166
167 var spot = CollectableSpots[index];
168 CollectableSpots.RemoveAt(index);
169 var room = Layout.Rooms[spot.Room];
170 room.Collectables.Add(spot.Id, id);
171
172 // Update the draw weights of neighboring spots
173 UpdateNeighborWeights(spot);
174 }

181 private void UpdateNeighborWeights(CollectableSpot spot)
182 {
183 var cluster = Clusters[spot.Room];
184
185 foreach (var other in CollectableSpots)
186 {
187 if (!cluster.Contains(other.Room))
188 continue;
189
190 if (other.Room == spot.Room)
191 {
192 // Spots are in the same room. Use distance between collectables.
193 var room = Layout.Rooms[spot.Room];
194 var distance = GetDistance(room, spot.Position, other.Position);
195 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
196 }
197 else
198 {
199 // Spots are in different rooms. Use total distance to connecting doors.
200 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
201 var fromRoom = Layout.Rooms[connection.FromRoom];
202 var toRoom = Layout.Rooms[connection.ToRoom];
203 var fromPosition = spot.Position;
204 var toPosition = other.Position;
205
206 if (connection.ToRoom == spot.Room)
207 (fromPosition, toPosition) = (toPosition, fromPosition);
208
209 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
210 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
211 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
212 }
213 }

252 private void AddRoomCollectableSpots(Room room)
253 {
254 var cells = room.Template.Cells;
255
256 for (int i = 0; i < cells.Rows; i++)
257 {
258 for (int j = 0; j < cells.Columns; j++)
259 {
260 var cell = cells[i, j];
261
262 if (cell == null)
263 continue;
264
265 foreach (var pair in cell.CollectableSpots.OrderBy(x => x.Key))
266 {
267 var position = new Vector2DInt(i, j);
268 var spot = new CollectableSpot(room.Id, position, pair.Key, pair.Value);
269 CollectableSpots.Add(spot);
270 }
271 }

312 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
313 {
314 if (!Distances.TryGetValue(room.Template, out var positions))
315 {
316 positions = new Dictionary<Vector2DInt, Array2D<int>>();
317 Distances.Add(room.Template, positions);
318 }
319
320 if (positions.TryGetValue(index1, out var distances))
321 return distances[index2.X, index2.Y];
322 if (positions.TryGetValue(index2, out distances))
323 return distances[index1.X, index1.Y];
324
325 distances = room.Template.FindCellDistances(index1);
326 positions.Add(index1, distances);
327 return distances[index2.X, index2.Y];
328 }
