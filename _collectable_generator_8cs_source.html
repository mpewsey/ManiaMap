138 private void AddCollectable(string group, int id)
139 {
140 // Draw a collectable spot and assign the collectable to the room.
141 var weights = GetWeights(group);
142 var index = RandomSeed.DrawWeightedIndex(weights);
143
144 if (index < 0)
145 throw new CollectableSpotNotFoundException($"Failed to find collectable spot for group: {group}.");
146
147 var spot = CollectableSpots[index];
148 CollectableSpots.RemoveAt(index);
149 var room = Layout.Rooms[spot.Room];
150 room.Collectables.Add(spot.Id, id);
151
152 // Update the draw weights of neighboring spots
153 UpdateNeighborWeights(spot);
154 }

161 private void UpdateNeighborWeights(CollectableSpot spot)
162 {
163 var cluster = Clusters[spot.Room];
164
165 foreach (var other in CollectableSpots)
166 {
167 if (!cluster.Contains(other.Room))
168 continue;
169
170 if (other.Room == spot.Room)
171 {
172 // Spots are in the same room. Use distance between collectables.
173 var room = Layout.Rooms[spot.Room];
174 var distance = GetDistance(room, spot.Position, other.Position);
175 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
176 }
177 else
178 {
179 // Spots are in different rooms. Use total distance to connecting doors.
180 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
181 var fromRoom = Layout.Rooms[connection.FromRoom];
182 var toRoom = Layout.Rooms[connection.ToRoom];
183 var fromPosition = spot.Position;
184 var toPosition = other.Position;
185
186 if (connection.ToRoom == spot.Room)
187 (fromPosition, toPosition) = (toPosition, fromPosition);
188
189 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
190 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
191 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
192 }
193 }

292 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
293 {
294 if (!Distances.TryGetValue(room.Template, out var positions))
295 {
296 positions = new Dictionary<Vector2DInt, Array2D<int>>();
297 Distances.Add(room.Template, positions);
298 }
299
300 if (positions.TryGetValue(index1, out var distances))
301 return distances[index2.X, index2.Y];
302 if (positions.TryGetValue(index2, out distances))
303 return distances[index1.X, index1.Y];
304
305 distances = room.Template.FindCellDistances(index1);
306 positions.Add(index1, distances);
307 return distances[index2.X, index2.Y];
308 }
