154 private void AddCollectable(string group, int id)
155 {
156 // Draw a collectable spot and assign the collectable to the room.
157 var weights = GetWeights(group);
158 var index = RandomSeed.DrawWeightedIndex(weights);
159
160 if (index < 0)
161 throw new CollectableSpotNotFoundException($"Failed to find collectable spot for group: {group}.");
162
163 var spot = CollectableSpots[index];
164 CollectableSpots.RemoveAt(index);
165 var room = Layout.Rooms[spot.Room];
166 room.Collectables.Add(spot.Id, id);
167
168 // Update the draw weights of neighboring spots
169 UpdateNeighborWeights(spot);
170 }

177 private void UpdateNeighborWeights(CollectableSpot spot)
178 {
179 var cluster = Clusters[spot.Room];
180
181 foreach (var other in CollectableSpots)
182 {
183 if (!cluster.Contains(other.Room))
184 continue;
185
186 if (other.Room == spot.Room)
187 {
188 // Spots are in the same room. Use distance between collectables.
189 var room = Layout.Rooms[spot.Room];
190 var distance = GetDistance(room, spot.Position, other.Position);
191 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
192 }
193 else
194 {
195 // Spots are in different rooms. Use total distance to connecting doors.
196 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
197 var fromRoom = Layout.Rooms[connection.FromRoom];
198 var toRoom = Layout.Rooms[connection.ToRoom];
199 var fromPosition = spot.Position;
200 var toPosition = other.Position;
201
202 if (connection.ToRoom == spot.Room)
203 (fromPosition, toPosition) = (toPosition, fromPosition);
204
205 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
206 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
207 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
208 }
209 }
210 }

248 private void AddRoomCollectableSpots(Room room)
249 {

252 for (int i = 0; i < cells.Rows; i++)
253 {
254 for (int j = 0; j < cells.Columns; j++)
255 {

308 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
309 {
310 if (!Distances.TryGetValue(room.Template, out var positions))
311 {
312 positions = new Dictionary<Vector2DInt, Array2D<int>>();
313 Distances.Add(room.Template, positions);
314 }
315
316 if (positions.TryGetValue(index1, out var distances))
317 return distances[index2.X, index2.Y];
318 if (positions.TryGetValue(index2, out distances))
319 return distances[index1.X, index1.Y];
320
321 distances = room.Template.FindCellDistances(index1);
322 positions.Add(index1, distances);
323 return distances[index2.X, index2.Y];
324 }
