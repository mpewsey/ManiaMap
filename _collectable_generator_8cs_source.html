161 private void AddCollectable(string group, int id)
162 {
163 // Draw a collectable spot and assign the collectable to the room.
164 var weights = GetWeights(group);
165 var index = RandomSeed.DrawWeightedIndex(weights);
166
167 if (index < 0)
168 throw new CollectableSpotNotFoundException($"Failed to find collectable spot for group: {group}.");
169
170 var spot = CollectableSpotWeights[index];
171 CollectableSpotWeights.RemoveAt(index);
172 var room = Layout.Rooms[spot.Room];
173 room.Collectables.Add(spot.Id, id);
174
175 // Update the draw weights of neighboring spots
176 UpdateNeighborWeights(spot);
177 }

184 private void UpdateNeighborWeights(CollectableSpotWeight spot)
185 {
186 var cluster = Clusters[spot.Room];
187
188 foreach (var other in CollectableSpotWeights)
189 {
190 if (!cluster.Contains(other.Room))
191 continue;
192
193 if (other.Room == spot.Room)
194 {
195 // Spots are in the same room. Use distance between collectables.
196 var room = Layout.Rooms[spot.Room];
197 var distance = GetDistance(room, spot.CollectableSpot.Position, other.CollectableSpot.Position);
198 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
199 }
200 else
201 {
202 // Spots are in different rooms. Use total distance to connecting doors.
203 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
204 var fromRoom = Layout.Rooms[connection.FromRoom];
205 var toRoom = Layout.Rooms[connection.ToRoom];
206 var fromPosition = spot.CollectableSpot.Position;
207 var toPosition = other.CollectableSpot.Position;
208
209 if (connection.ToRoom == spot.Room)
210 (fromPosition, toPosition) = (toPosition, fromPosition);
211
212 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
213 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
214 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
215 }
216 }
217 }

223 private double[] GetWeights(string group)
224 {
225 var weights = new double[CollectableSpotWeights.Count];
226
227 for (int i = 0; i < weights.Length; i++)
228 {
229 var spot = CollectableSpotWeights[i];
230
231 if (spot.CollectableSpot.Group == group)
232 weights[i] = spot.GetWeight(DoorPower, NeighborPower);
233 }

255 private void CreateRoomCollectableSpotWeights(Room room)

258
259 foreach (var pair in room.Template.CollectableSpots.OrderBy(x => x.Key))

261 if (cells.GetOrDefault(pair.Value.Position) != null)
262 CollectableSpotWeights.Add(new CollectableSpotWeight(room.Id, pair.Key, pair.Value));

303 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
304 {
305 if (!Distances.TryGetValue(room.Template, out var positions))
306 {
307 positions = new Dictionary<Vector2DInt, Array2D<int>>();
308 Distances.Add(room.Template, positions);
309 }
310
311 if (positions.TryGetValue(index1, out var distances))
312 return distances[index2.X, index2.Y];
313 if (positions.TryGetValue(index2, out distances))
314 return distances[index1.X, index1.Y];
315
316 distances = room.Template.FindCellDistances(index1);
317 positions.Add(index1, distances);
318 return distances[index2.X, index2.Y];
319 }
