112 private void AddCollectable(string group, int id)
113 {
114 // Draw a collectable spot and assign the collectable to the room.
115 var weights = GetWeights(group);
116 var index = RandomSeed.DrawWeightedIndex(weights);
117
118 if (index < 0)
119 throw new Exception($"Failed to find collectable spot for group: {group}.");
120
121 var spot = CollectableSpots[index];
122 CollectableSpots.RemoveAt(index);
123 var room = Layout.Rooms[spot.Room];
124 room.Collectables.Add(spot.Position, id);
125
126 // Update the draw weights of neighboring spots
127 UpdateNeighborWeights(spot);
128 }

135 private void UpdateNeighborWeights(CollectableSpot spot)
136 {
137 var cluster = Clusters[spot.Room];
138
139 foreach (var other in CollectableSpots)
140 {
141 if (!cluster.Contains(other.Room))
142 continue;
143
144 if (other.Room == spot.Room)
145 {
146 // Spots are in the same room. Use distance between collectables.
147 var room = Layout.Rooms[spot.Room];
148 var distance = GetDistance(room, spot.Position, other.Position);
149 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
150 }
151 else
152 {
153 // Spots are in different rooms. Use total distance to connecting doors.
154 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
155 var fromRoom = Layout.Rooms[connection.FromRoom];
156 var toRoom = Layout.Rooms[connection.ToRoom];
157 var fromPosition = spot.Position;
158 var toPosition = other.Position;
159
160 if (connection.ToRoom == spot.Room)
161 (fromPosition, toPosition) = (toPosition, fromPosition);
162
163 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
164 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
165 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
166 }
167 }
168 }

262 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
263 {
264 if (!Distances.TryGetValue(room.Template, out var positions))
265 {
266 positions = new Dictionary<Vector2DInt, Array2D<int>>();
267 Distances.Add(room.Template, positions);
268 }
269
270 if (positions.TryGetValue(index1, out var distances))
271 return distances[index2.X, index2.Y];
272 if (positions.TryGetValue(index2, out distances))
273 return distances[index1.X, index1.Y];
274
275 distances = room.Template.FindCellDistances(index1);
276 positions.Add(index1, distances);
277 return distances[index2.X, index2.Y];
278 }
