150 private void AddCollectable(string group, int id)
151 {
152 // Draw a collectable spot and assign the collectable to the room.
153 var weights = GetWeights(group);
154 var index = RandomSeed.DrawWeightedIndex(weights);
155
156 if (index < 0)
157 throw new CollectableSpotNotFoundException($"Failed to find collectable spot for group: {group}.");
158
159 var spot = CollectableSpots[index];
160 CollectableSpots.RemoveAt(index);
161 var room = Layout.Rooms[spot.Room];
162 room.Collectables.Add(spot.Id, id);
163
164 // Update the draw weights of neighboring spots
165 UpdateNeighborWeights(spot);
166 }

173 private void UpdateNeighborWeights(CollectableSpot spot)
174 {
175 var cluster = Clusters[spot.Room];
176
177 foreach (var other in CollectableSpots)
178 {
179 if (!cluster.Contains(other.Room))
180 continue;
181
182 if (other.Room == spot.Room)
183 {
184 // Spots are in the same room. Use distance between collectables.
185 var room = Layout.Rooms[spot.Room];
186 var distance = GetDistance(room, spot.Position, other.Position);
187 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
188 }
189 else
190 {
191 // Spots are in different rooms. Use total distance to connecting doors.
192 var connection = Layout.GetDoorConnection(spot.Room, other.Room);
193 var fromRoom = Layout.Rooms[connection.FromRoom];
194 var toRoom = Layout.Rooms[connection.ToRoom];
195 var fromPosition = spot.Position;
196 var toPosition = other.Position;
197
198 if (connection.ToRoom == spot.Room)
199 (fromPosition, toPosition) = (toPosition, fromPosition);
200
201 var distance = GetDistance(fromRoom, connection.FromDoor.Position, fromPosition);
202 distance += GetDistance(toRoom, connection.ToDoor.Position, toPosition);
203 other.NeighborWeight = Math.Min(distance, other.NeighborWeight);
204 }
205 }

245 {

304 private int GetDistance(Room room, Vector2DInt index1, Vector2DInt index2)
305 {
306 if (!Distances.TryGetValue(room.Template, out var positions))
307 {
308 positions = new Dictionary<Vector2DInt, Array2D<int>>();
309 Distances.Add(room.Template, positions);
310 }
311
312 if (positions.TryGetValue(index1, out var distances))
313 return distances[index2.X, index2.Y];
314 if (positions.TryGetValue(index2, out distances))
315 return distances[index1.X, index1.Y];
316
317 distances = room.Template.FindCellDistances(index1);
318 positions.Add(index1, distances);
319 return distances[index2.X, index2.Y];
320 }
