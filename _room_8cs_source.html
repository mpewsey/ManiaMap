145 public Vector2DInt VisibleCellCount(RoomState state)
146 {
147 var cellCount = 0;
148 var visibleCount = 0;
149 var cells = Template.Cells;
150 var visibleCells = state.VisibleCells;
151
152 for (int i = 0; i < cells.Rows; i++)
153 {
154 for (int j = 0; j < cells.Columns; j++)
155 {
156 if (cells[i, j] != null)
157 {
158 if (visibleCells[i, j])
159 visibleCount++;
160 cellCount++;
161 }
162 }
163 }
