35 public List<List<int>> FindBranches(LayoutGraph graph)
36 {
37 Graph = graph;
38 Marked.Clear();
39 Parents.Clear();
40 Branches = new List<List<int>>();
41 var cycles = graph.FindCycles();
42
43 // Add trunk nodes to marked set.
44 foreach (var cycle in cycles)
45 {
46 foreach (var node in cycle)
47 {
48 Marked.Add(node);
49 }
50 }
51
52 // Use the node with the max neighbors if cycles do not exist.
53 if (Marked.Count == 0)
54 {
55 Marked.Add(graph.MaxNeighborNode());
56 }
57
58 // Search for branches beginning at each trunk node.
59 var trunk = Marked.ToArray();
60 Array.Sort(trunk);
61
62 foreach (var node in trunk)
63 {
64 BranchSearch(node, -1);
65 }
66
67 return Branches;
68 }

75 private void BranchSearch(int node, int parent)
76 {
77 // If the node already has a parent, then it has been traversed previously from another trunk node.
78 // The node is a member of a connecting branch between two trunk nodes.
79 if (Parents.ContainsKey(node))
80 {
81 var branch = new List<int> { parent, node };
82 AddParentsToBranch(node, branch);
83 Branches.Add(branch);
84 return;
85 }
86
87 var neighbors = Graph.GetNeighbors(node);
88 Parents[node] = parent;
89
90 // If the node is a deadend node, accumulate the parents into a branch.
91 if (neighbors.Count == 1 && !Marked.Contains(node))
92 {
93 var branch = new List<int> { node };
94 AddParentsToBranch(node, branch);
95 Branches.Add(branch);
96 return;
97 }
98
99 foreach (var neighbor in neighbors)
100 {
101 if (!Marked.Contains(neighbor))
102 BranchSearch(neighbor, node);
103 }
104 }

111 private void AddParentsToBranch(int node, List<int> branch)
112 {
113 var current = node;
114 var isMarked = false;
115 Marked.Add(node);
116
117 // Accumulate parents into the branch until a marked node is encountered.
118 while (!isMarked)
119 {
120 current = Parents[current];
121 branch.Add(current);
122 isMarked = !Marked.Add(current);
123 }
124 }
