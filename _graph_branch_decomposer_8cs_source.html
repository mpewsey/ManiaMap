51 public static List<List<int>> FindBranches(LayoutGraph graph)

98 private static void BranchSearch(Data data, int node, int parent)
99 {
100 // If the node already has a parent, then it has been traversed previously from another trunk node.
101 // The node is a member of a connecting branch between two trunk nodes.
102 if (data.Parents.ContainsKey(node))
103 {
104 var branch = new List<int> { parent, node };
105 AddParentsToBranch(data, node, branch);
106 data.Branches.Add(branch);
107 return;
108 }
109
110 var neighbors = data.Graph.GetNeighbors(node);
111 data.Parents[node] = parent;
112
113 // If the node is a deadend node, accumulate the parents into a branch.
114 if (neighbors.Count == 1 && !data.Marked.Contains(node))
115 {
116 var branch = new List<int> { node };
117 AddParentsToBranch(data, node, branch);
118 data.Branches.Add(branch);
119 return;
120 }
121
122 foreach (var neighbor in neighbors)
123 {
124 if (!data.Marked.Contains(neighbor))
125 BranchSearch(data, neighbor, node);
126 }
127 }

135 private static void AddParentsToBranch(Data data, int node, List<int> branch)
136 {
137 var current = node;
138 var isMarked = false;
139 data.Marked.Add(node);
140
141 // Accumulate parents into the branch until a marked node is encountered.
142 while (!isMarked)
143 {
144 current = data.Parents[current];
145 branch.Add(current);
146 isMarked = !data.Marked.Add(current);
147 }
148 }
