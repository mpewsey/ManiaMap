110 private void BranchSearch(T node, T parent)
111 {
112 // If the node already has a parent, then it has been traversed previously from another trunk node.
113 // The node is a member of a connecting branch between two trunk nodes.
114 if (Parents.ContainsKey(node))
115 {
116 var branch = new List<T> { parent, node };
117 AddParentsToBranch(node, branch);
118 Branches.Add(branch);
119 return;
120 }
121
122 var neighbors = Neighbors[node];
123 Parents[node] = parent;
124
125 // If the node is a deadend node, accumulate the parents into a branch.
126 if (neighbors.Count == 1 && !Marked.Contains(node))
127 {
128 var branch = new List<T> { node };
129 AddParentsToBranch(node, branch);
130 Branches.Add(branch);
131 return;
132 }
133
134 foreach (var neighbor in neighbors)
135 {
136 if (!Marked.Contains(neighbor))
137 BranchSearch(neighbor, node);
138 }
139 }

146 private void AddParentsToBranch(T node, List<T> branch)
147 {
148 var current = node;
149 var isMarked = false;
150 Marked.Add(node);
151
152 // Accumulate parents into the branch until a marked node is encountered.
153 while (!isMarked)
154 {
155 current = Parents[current];
156 branch.Add(current);
157 isMarked = !Marked.Add(current);
158 }
159 }
