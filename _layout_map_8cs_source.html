127 public Dictionary<int, Image> CreateImages(Layout layout, LayoutState state = null)
128 {
129 Layout = layout;
130 LayoutState = state;
131 LayoutBounds = Layout.GetBounds();
132 RoomDoors = Layout.GetRoomDoors();
133
134 var width = TileSize.X * (Padding.Left + Padding.Right + LayoutBounds.Width);
135 var height = TileSize.Y * (Padding.Top + Padding.Bottom + LayoutBounds.Height);
136 var dict = new Dictionary<int, Image>();
137
138 foreach (var room in Layout.Rooms.Values)
139 {
140 if (!dict.ContainsKey(room.Position.Z))
141 {
142 var map = new Image<Rgba32>(width, height);
143 map.Mutate(x => DrawMap(x, room.Position.Z));
144 dict.Add(room.Position.Z, map);
145 }
146 }
147
148 return dict;
149 }

188 private void DrawMapTiles(IImageProcessingContext image, int z)
189 {
190 foreach (var room in Layout.Rooms.Values)
191 {
192 // If room Z (layer) value is not equal, go to next room.
193 if (room.Position.Z != z)
194 continue;
195
196 var roomState = LayoutState?.RoomStates[room.Id];
197 var cells = room.Template.Cells;
198 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * TileSize.X;
199 var y0 = (room.Position.X - LayoutBounds.Y + Padding.Top) * TileSize.Y;
200 var cellTile = new Image<Rgba32>(TileSize.X, TileSize.Y);
201 cellTile.Mutate(x => x.BackgroundColor(ConvertColor(room.Color)));
202
203 for (int i = 0; i < cells.Rows; i++)
204 {
205 for (int j = 0; j < cells.Columns; j++)
206 {
207 var cell = cells[i, j];
208 var position = new Vector2DInt(i, j);
209
210 // If cell it empty, go to next cell.
211 if (cell == null)
212 continue;
213
214 // If room state is defined and is not visible, go to next cell.
215 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
216 continue;
217
218 // Calculate draw position
219 var x = TileSize.X * j + x0;
220 var y = TileSize.Y * i + y0;
221 var point = new Point(x, y);
222
223 // Get adjacent cells
224 var north = cells.GetOrDefault(i - 1, j);
225 var south = cells.GetOrDefault(i + 1, j);
226 var west = cells.GetOrDefault(i, j - 1);
227 var east = cells.GetOrDefault(i, j + 1);
228
229 // Get the wall or door tiles
230 var topTile = GetTile(room, position, DoorDirection.Top, cell.TopDoor, null, "TopDoor", null);
231 var bottomTile = GetTile(room, position, DoorDirection.Bottom, cell.BottomDoor, null, "BottomDoor", null);
232 var northTile = GetTile(room, position, DoorDirection.North, cell.NorthDoor, north, "NorthDoor", "NorthWall");
233 var southTile = GetTile(room, position, DoorDirection.South, cell.SouthDoor, south, "SouthDoor", "SouthWall");
234 var westTile = GetTile(room, position, DoorDirection.West, cell.WestDoor, west, "WestDoor", "WestWall");
235 var eastTile = GetTile(room, position, DoorDirection.East, cell.EastDoor, east, "EastDoor", "EastWall");
236
237 // Add cell background fill
238 image.DrawImage(cellTile, point, 1);
239
240 // Superimpose applicable map tiles
241 if (northTile != null)
242 image.DrawImage(northTile, point, 1);
243 if (southTile != null)
244 image.DrawImage(southTile, point, 1);
245 if (westTile != null)
246 image.DrawImage(westTile, point, 1);
247 if (eastTile != null)
248 image.DrawImage(eastTile, point, 1);
249 if (topTile != null)
250 image.DrawImage(topTile, point, 1);
251 if (bottomTile != null)
252 image.DrawImage(bottomTile, point, 1);
253 }
254 }
255 }
