141 public Dictionary<int, Image> CreateImages(Layout layout, LayoutState state = null)
142 {
143 Initialize(layout, state);
144
145 var width = TileSize.X * (Padding.Left + Padding.Right + LayoutBounds.Width);
146 var height = TileSize.Y * (Padding.Top + Padding.Bottom + LayoutBounds.Height);
147 var dict = new Dictionary<int, Image>();
148
149 foreach (var room in Layout.Rooms.Values)
150 {
151 if (!dict.ContainsKey(room.Position.Z))
152 {
153 var map = new Image<Rgba32>(width, height);
154 map.Mutate(x => DrawMap(x, room.Position.Z));
155 dict.Add(room.Position.Z, map);
156 }
157 }
158
159 return dict;
160 }

199 private void DrawMapTiles(IImageProcessingContext image, int z)
200 {
201 foreach (var room in Layout.Rooms.Values)
202 {
203 // If room Z (layer) value is not equal, go to next room.
204 if (room.Position.Z != z)
205 continue;
206
207 var roomState = LayoutState?.RoomStates[room.Id];
208 var cells = room.Template.Cells;
209 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * TileSize.X;
210 var y0 = (room.Position.X - LayoutBounds.Y + Padding.Top) * TileSize.Y;
211 var cellTile = new Image<Rgba32>(TileSize.X, TileSize.Y);
212 cellTile.Mutate(x => x.BackgroundColor(ConvertColor(room.Color)));
213
214 for (int i = 0; i < cells.Rows; i++)
215 {
216 for (int j = 0; j < cells.Columns; j++)
217 {
218 var cell = cells[i, j];
219 var position = new Vector2DInt(i, j);
220
221 // If cell is empty, go to next cell.
222 if (cell == null)
223 continue;
224
225 // If room state is defined and is not visible, go to next cell.
226 if (roomState != null && !roomState.CellIsVisible(position))
227 continue;
228
229 // Calculate draw position
230 var x = TileSize.X * j + x0;
231 var y = TileSize.Y * i + y0;
232 var point = new Point(x, y);
233
234 // Get adjacent cells
235 var north = cells.GetOrDefault(i - 1, j);
236 var south = cells.GetOrDefault(i + 1, j);
237 var west = cells.GetOrDefault(i, j - 1);
238 var east = cells.GetOrDefault(i, j + 1);
239
240 // Get the wall or door tiles
241 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
242 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
243 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
244 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
245 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
246 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
247
248 // Add cell background fill
249 image.DrawImage(cellTile, point, 1);
250
251 // Superimpose applicable map tiles
252 if (northTile != null)
253 image.DrawImage(northTile, point, 1);
254 if (southTile != null)
255 image.DrawImage(southTile, point, 1);
256 if (westTile != null)
257 image.DrawImage(westTile, point, 1);
258 if (eastTile != null)
259 image.DrawImage(eastTile, point, 1);
260 if (topTile != null)
261 image.DrawImage(topTile, point, 1);
262 if (bottomTile != null)
263 image.DrawImage(bottomTile, point, 1);
264
265 DrawFeatureTiles(image, cell, point);
266 }
267 }
268 }

307 private Image GetTile(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
308 {
309 if (cell.GetDoor(direction) != null && DoorExists(room, position, direction))
310 return Tiles[MapTileType.GetDoorTileType(direction)];
311
312 var wallType = MapTileType.GetWallTileType(direction);
313
314 if (wallType != MapTileType.None && neighbor == null)
315 return Tiles[wallType];
316
317 return null;
318 }
