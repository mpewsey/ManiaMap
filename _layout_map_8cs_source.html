195 private void DrawMapTiles(IImageProcessingContext image, int z)
196 {
197 var tileSize = Settings.TileSize;
198 var padding = Settings.Padding;
199
200 foreach (var room in Layout.Rooms.Values)
201 {
202 // If room Z (layer) value is not equal, go to next room.
203 if (room.Position.Z != z)
204 continue;
205
206 var roomState = LayoutState?.RoomStates[room.Id];
207 var cells = room.Template.Cells;
208 var x0 = (room.Position.Y - LayoutBounds.X + padding.Left) * tileSize.X;
209 var y0 = (room.Position.X - LayoutBounds.Y + padding.Top) * tileSize.Y;
210 var cellTile = new Image<Rgba32>(tileSize.X, tileSize.Y);
211 cellTile.Mutate(x => x.BackgroundColor(ConvertColor(room.Color)));
212
213 for (int i = 0; i < cells.Rows; i++)
214 {
215 for (int j = 0; j < cells.Columns; j++)
216 {
217 var cell = cells[i, j];
218 var position = new Vector2DInt(i, j);
219
220 // If cell it empty, go to next cell.
221 if (cell == null)
222 continue;
223
224 // If room state is defined and is not visible, go to next cell.
225 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
226 continue;
227
228 // Calculate draw position
229 var x = tileSize.X * j + x0;
230 var y = tileSize.Y * i + y0;
231 var point = new Point(x, y);
232
233 // Get adjacent cells
234 var north = cells.GetOrDefault(i - 1, j);
235 var south = cells.GetOrDefault(i + 1, j);
236 var west = cells.GetOrDefault(i, j - 1);
237 var east = cells.GetOrDefault(i, j + 1);
238
239 // Get the wall or door tiles
240 var topTile = GetTile(room, position, DoorDirection.Top, cell.TopDoor, null, "TopDoor", null);
241 var bottomTile = GetTile(room, position, DoorDirection.Bottom, cell.BottomDoor, null, "BottomDoor", null);
242 var northTile = GetTile(room, position, DoorDirection.North, cell.NorthDoor, north, "NorthDoor", "NorthWall");
243 var southTile = GetTile(room, position, DoorDirection.South, cell.SouthDoor, south, "SouthDoor", "SouthWall");
244 var westTile = GetTile(room, position, DoorDirection.West, cell.WestDoor, west, "WestDoor", "WestWall");
245 var eastTile = GetTile(room, position, DoorDirection.East, cell.EastDoor, east, "EastDoor", "EastWall");
246
247 // Add cell background fill
248 image.DrawImage(cellTile, point, 1);
249
250 // Superimpose applicable map tiles
251 if (northTile != null)
252 image.DrawImage(northTile, point, 1);
253 if (southTile != null)
254 image.DrawImage(southTile, point, 1);
255 if (westTile != null)
256 image.DrawImage(westTile, point, 1);
257 if (eastTile != null)
258 image.DrawImage(eastTile, point, 1);
259 if (topTile != null)
260 image.DrawImage(topTile, point, 1);
261 if (bottomTile != null)
262 image.DrawImage(bottomTile, point, 1);
263 }
264 }
265 }
