157 public Dictionary<int, Image> CreateImages(Layout layout, LayoutState state = null)
158 {
159 Initialize(layout, state);
160
161 var width = TileSize.X * (Padding.Left + Padding.Right + LayoutBounds.Width);
162 var height = TileSize.Y * (Padding.Top + Padding.Bottom + LayoutBounds.Height);
163 var dict = new Dictionary<int, Image>();
164
165 foreach (var room in Layout.Rooms.Values)
166 {
167 if (!dict.ContainsKey(room.Position.Z))
168 {
169 var map = new Image<Rgba32>(width, height);
170 map.Mutate(x => DrawMap(x, room.Position.Z));
171 dict.Add(room.Position.Z, map);
172 }
173 }
174
175 return dict;
176 }

215 private void DrawMapTiles(IImageProcessingContext image, int z)
216 {
217 var roomTile = new Image<Rgba32>(TileSize.X, TileSize.Y);
218 roomTile.Mutate(x => x.BackgroundColor(ConvertColor(RoomColor)));
219
220 foreach (var room in Layout.Rooms.Values)
221 {
222 // If room Z (layer) value is not equal, go to next room.
223 if (room.Position.Z != z)
224 continue;
225
226 var roomState = LayoutState?.RoomStates[room.Id];
227 var cells = room.Template.Cells;
228 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * TileSize.X;
229 var y0 = (room.Position.X - LayoutBounds.Y + Padding.Top) * TileSize.Y;
230 var cellTile = new Image<Rgba32>(TileSize.X, TileSize.Y);
231 cellTile.Mutate(x => x.BackgroundColor(ConvertColor(room.Color)));
232
233 for (int i = 0; i < cells.Rows; i++)
234 {
235 for (int j = 0; j < cells.Columns; j++)
236 {
237 var cell = cells[i, j];
238
239 // If cell is empty, go to next cell.
240 if (cell == null)
241 continue;
242
243 var position = new Vector2DInt(i, j);
244 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
245
246 // If room state is defined and is not visible, go to next cell.
247 if (!isCompletelyVisible && !roomState.IsVisible)
248 continue;
249
250 // Calculate draw position
251 var x = TileSize.X * j + x0;
252 var y = TileSize.Y * i + y0;
253 var point = new Point(x, y);
254
255 // Get adjacent cells
256 var north = cells.GetOrDefault(i - 1, j);
257 var south = cells.GetOrDefault(i + 1, j);
258 var west = cells.GetOrDefault(i, j - 1);
259 var east = cells.GetOrDefault(i, j + 1);
260
261 // Get the wall or door tiles
262 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
263 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
264 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
265 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
266 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
267 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
268
269 // Add cell background fill
270 var tile = isCompletelyVisible ? cellTile : roomTile;
271 image.DrawImage(tile, point, 1);
272
273 // Superimpose applicable map tiles
274 if (northTile != null)
275 image.DrawImage(northTile, point, 1);
276 if (southTile != null)
277 image.DrawImage(southTile, point, 1);
278 if (westTile != null)
279 image.DrawImage(westTile, point, 1);
280 if (eastTile != null)
281 image.DrawImage(eastTile, point, 1);
282 if (topTile != null)
283 image.DrawImage(topTile, point, 1);
284 if (bottomTile != null)
285 image.DrawImage(bottomTile, point, 1);
286
287 // Draw features only if the cell is completely visible
288 if (isCompletelyVisible)
289 DrawFeatureTiles(image, cell, point);
290 }
291 }
292 }

330 private Image GetTile(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
331 {
332 if (Door.ShowDoor(DoorDrawMode, direction) && cell.GetDoor(direction) != null && DoorExists(room, position, direction))
333 return GetMapTile(MapTileType.GetDoorTileType(direction));
334
335 if (neighbor == null)
336 return GetMapTile(MapTileType.GetWallTileType(direction));
337
338 return null;
339 }
