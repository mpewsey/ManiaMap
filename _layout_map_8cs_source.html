156 public Dictionary<int, Image> CreateImages(Layout layout, LayoutState state = null)
157 {
158 Initialize(layout, state);
159
160 var width = TileSize.X * (Padding.Left + Padding.Right + LayoutBounds.Width);
161 var height = TileSize.Y * (Padding.Top + Padding.Bottom + LayoutBounds.Height);
162 var dict = new Dictionary<int, Image>();
163
164 foreach (var room in Layout.Rooms.Values)
165 {
166 if (!dict.ContainsKey(room.Position.Z))
167 {
168 var map = new Image<Rgba32>(width, height);
169 map.Mutate(x => DrawMap(x, room.Position.Z));
170 dict.Add(room.Position.Z, map);
171 }
172 }
173
174 return dict;
175 }

214 private void DrawMapTiles(IImageProcessingContext image, int z)
215 {
216 var roomTile = new Image<Rgba32>(TileSize.X, TileSize.Y);
217 roomTile.Mutate(x => x.BackgroundColor(ConvertColor(RoomColor)));
218
219 foreach (var room in Layout.Rooms.Values)
220 {
221 // If room Z (layer) value is not equal, go to next room.
222 if (room.Position.Z != z)
223 continue;
224
225 var roomState = LayoutState?.RoomStates[room.Id];
226 var cells = room.Template.Cells;
227 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * TileSize.X;
228 var y0 = (room.Position.X - LayoutBounds.Y + Padding.Top) * TileSize.Y;
229 var cellTile = new Image<Rgba32>(TileSize.X, TileSize.Y);
230 cellTile.Mutate(x => x.BackgroundColor(ConvertColor(room.Color)));
231
232 for (int i = 0; i < cells.Rows; i++)
233 {
234 for (int j = 0; j < cells.Columns; j++)
235 {
236 var cell = cells[i, j];
237
238 // If cell is empty, go to next cell.
239 if (cell == null)
240 continue;
241
242 var position = new Vector2DInt(i, j);
243 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
244
245 // If room state is defined and is not visible, go to next cell.
246 if (!isCompletelyVisible && !roomState.IsVisible)
247 continue;
248
249 // Calculate draw position
250 var x = TileSize.X * j + x0;
251 var y = TileSize.Y * i + y0;
252 var point = new Point(x, y);
253
254 // Get adjacent cells
255 var north = cells.GetOrDefault(i - 1, j);
256 var south = cells.GetOrDefault(i + 1, j);
257 var west = cells.GetOrDefault(i, j - 1);
258 var east = cells.GetOrDefault(i, j + 1);
259
260 // Get the wall or door tiles
261 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
262 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
263 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
264 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
265 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
266 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
267
268 // Add cell background fill
269 var tile = isCompletelyVisible ? cellTile : roomTile;
270 image.DrawImage(tile, point, 1);
271
272 // Superimpose applicable map tiles
273 if (northTile != null)
274 image.DrawImage(northTile, point, 1);
275 if (southTile != null)
276 image.DrawImage(southTile, point, 1);
277 if (westTile != null)
278 image.DrawImage(westTile, point, 1);
279 if (eastTile != null)
280 image.DrawImage(eastTile, point, 1);
281 if (topTile != null)
282 image.DrawImage(topTile, point, 1);
283 if (bottomTile != null)
284 image.DrawImage(bottomTile, point, 1);
285
286 // Draw features only if the cell is completely visible
287 if (isCompletelyVisible)
288 DrawFeatureTiles(image, cell, point);
289 }
290 }
291 }
292 }

329 private Image GetTile(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
330 {
331 if (Door.ShowDoor(DoorDrawMode, direction) && cell.GetDoor(direction) != null && DoorExists(room, position, direction))
332 return GetMapTile(MapTileType.GetDoorTileType(direction));
333
334 if (neighbor == null)
335 return GetMapTile(MapTileType.GetWallTileType(direction));
336
337 return null;
