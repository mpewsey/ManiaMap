183 private void ApplyVariations(RandomSeed randomSeed)
184 {
185 foreach (var pair in NodeVariations.OrderBy(x => x.Key))
186 {
187 var variations = randomSeed.Shuffled(pair.Value);
188
189 for (int i = 0; i < variations.Count; i++)
190 {
191 SwapEdges(pair.Value[i], variations[i]);
192 }
193 }
194 }

295 public void SwapEdges(int id1, int id2)
296 {
297 if (id1 == id2)
298 return;
299
300 // Get a set of all edges with the nodes.
301 var edges = GetEdges(id1).Concat(GetEdges(id2)).Distinct().ToList();
302
303 // Remove existing edges
304 foreach (var edge in edges)
305 {
306 RemoveEdge(edge.FromNode, edge.ToNode);
307 }
308
309 // Create new edges with the node replacements and copy properties to new edges.
310 foreach (var edge in edges)
311 {
312 var node1 = edge.FromNode == id1 ? id2 : edge.FromNode == id2 ? id1 : edge.FromNode;
313 var node2 = edge.ToNode == id1 ? id2 : edge.ToNode == id2 ? id1 : edge.ToNode;
314 AddEdge(node1, node2).SetProperties(edge);
315 }
316 }

346 public LayoutEdge AddEdge(int fromNode, int toNode)
347 {
348 if (fromNode == toNode)
349 throw new InvalidIdException($"From node cannot be the same as to node: {fromNode}.");
350
351 if (!TryGetEdge(fromNode, toNode, out var edge))
352 {
353 edge = new LayoutEdge(fromNode, toNode);
354 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
355 }
356
357 AddNode(fromNode);
358 AddNode(toNode);
359
360 var fromNeighbors = Neighbors[fromNode];
361 var toNeighbors = Neighbors[toNode];
362
363 if (!fromNeighbors.Contains(toNode))
364 fromNeighbors.Add(toNode);
365
366 if (!toNeighbors.Contains(fromNode))
367 toNeighbors.Add(fromNode);
368
369 return edge;
370 }

376 public void RemoveNode(int id)
377 {
378 var neighbors = Neighbors[id];
379
380 for (int i = neighbors.Count - 1; i >= 0; i--)
381 {
382 var neighbor = neighbors[i];
383 Neighbors[neighbor].Remove(id);
384
385 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
386 Edges.Remove(new EdgeIndexes(neighbor, id));
387 }
388
389 Neighbors.Remove(id);
390 Nodes.Remove(id);
391 RemoveNodeVariations(id);
392 }

400 public void RemoveEdge(int node1, int node2)
401 {
402 Neighbors[node1].Remove(node2);
403 Neighbors[node2].Remove(node1);
404
405 if (!Edges.Remove(new EdgeIndexes(node1, node2)))
406 Edges.Remove(new EdgeIndexes(node2, node1));
