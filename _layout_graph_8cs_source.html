96 public LayoutEdge AddEdge(int fromNode, int toNode)
97 {
98 if (!TryGetEdge(fromNode, toNode, out var edge))
99 {
100 edge = new LayoutEdge(fromNode, toNode);
101 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
102 }
103
104 AddNode(fromNode);
105 AddNode(toNode);
106
107 var fromNeighbors = Neighbors[fromNode];
108 var toNeighbors = Neighbors[toNode];
109
110 if (!fromNeighbors.Contains(toNode))
111 fromNeighbors.Add(toNode);
112
113 if (!toNeighbors.Contains(fromNode))
114 toNeighbors.Add(fromNode);
115
116 return edge;
117 }

123 public void RemoveNode(int id)
124 {
125 var neighbors = Neighbors[id];
126
127 for (int i = neighbors.Count - 1; i >= 0; i--)
128 {
129 var neighbor = neighbors[i];
130 Neighbors[neighbor].Remove(id);
131
132 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
133 Edges.Remove(new EdgeIndexes(neighbor, id));
134 }
135
136 Neighbors.Remove(id);
137 Nodes.Remove(id);
138 }
