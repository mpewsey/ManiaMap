121 private void ApplyVariations(RandomSeed randomSeed)
122 {
123 foreach (var pair in NodeVariations.OrderBy(x => x.Key))
124 {
125 var variations = randomSeed.Shuffled(pair.Value);
126
127 for (int i = 0; i < variations.Count; i++)
128 {
129 SwapEdges(pair.Value[i], variations[i]);
130 }
131 }
132 }

233 public void SwapEdges(int id1, int id2)
234 {
235 if (id1 == id2)
236 return;
237
238 // Get a set of all edges with the nodes.
239 var edges = GetEdges(id1).Concat(GetEdges(id2)).Distinct().ToList();
240
241 // Remove existing edges
242 foreach (var edge in edges)
243 {
244 RemoveEdge(edge.FromNode, edge.ToNode);
245 }
246
247 // Create new edges with the node replacements and copy properties to new edges.
248 foreach (var edge in edges)
249 {
250 var node1 = edge.FromNode == id1 ? id2 : edge.FromNode == id2 ? id1 : edge.FromNode;
251 var node2 = edge.ToNode == id1 ? id2 : edge.ToNode == id2 ? id1 : edge.ToNode;
252 AddEdge(node1, node2).SetProperties(edge);
253 }
254 }

284 public LayoutEdge AddEdge(int fromNode, int toNode)
285 {
286 if (!TryGetEdge(fromNode, toNode, out var edge))
287 {
288 edge = new LayoutEdge(fromNode, toNode);
289 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
290 }
291
292 AddNode(fromNode);
293 AddNode(toNode);
294
295 var fromNeighbors = Neighbors[fromNode];
296 var toNeighbors = Neighbors[toNode];
297
298 if (!fromNeighbors.Contains(toNode))
299 fromNeighbors.Add(toNode);
300
301 if (!toNeighbors.Contains(fromNode))
302 toNeighbors.Add(fromNode);
303
304 return edge;
305 }

311 public void RemoveNode(int id)
312 {
313 var neighbors = Neighbors[id];
314
315 for (int i = neighbors.Count - 1; i >= 0; i--)
316 {
317 var neighbor = neighbors[i];
318 Neighbors[neighbor].Remove(id);
319
320 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
321 Edges.Remove(new EdgeIndexes(neighbor, id));
322 }
323
324 Neighbors.Remove(id);
325 Nodes.Remove(id);
326 RemoveNodeVariations(id);
327 }

335 public void RemoveEdge(int node1, int node2)

338 Neighbors[node2].Remove(node1);

340 if (!Edges.Remove(new EdgeIndexes(node1, node2)))
341 Edges.Remove(new EdgeIndexes(node2, node1));
