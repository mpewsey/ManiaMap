205 private void ApplyVariations(RandomSeed randomSeed)
206 {
207 foreach (var pair in NodeVariations.OrderBy(x => x.Key))
208 {
209 var variations = randomSeed.Shuffled(pair.Value);
210
211 for (int i = 0; i < variations.Count; i++)
212 {
213 SwapEdges(pair.Value[i], variations[i]);
214 }
215 }

328 public void SwapEdges(int id1, int id2)
329 {
330 if (id1 != id2)
331 {
332 // Get a set of all edges with the nodes.
333 var edges = GetEdges(id1).Concat(GetEdges(id2)).Distinct().ToList();
334
335 // Remove existing edges
336 foreach (var edge in edges)
337 {
338 RemoveEdge(edge.FromNode, edge.ToNode);
339 }
340
341 // Create new edges with the node replacements and copy properties to new edges.
342 foreach (var edge in edges)
343 {
344 var node1 = edge.FromNode == id1 ? id2 : edge.FromNode == id2 ? id1 : edge.FromNode;
345 var node2 = edge.ToNode == id1 ? id2 : edge.ToNode == id2 ? id1 : edge.ToNode;
346 AddEdge(node1, node2).SetProperties(edge);
347 }
348 }

379 public LayoutEdge AddEdge(int fromNode, int toNode)
380 {
381 if (fromNode == toNode)
382 throw new InvalidIdException($"From node cannot be the same as to node: {fromNode}.");
383
384 // Add edge if it doesn't already exist.
385 if (!TryGetEdge(fromNode, toNode, out var edge))
386 {
387 edge = new LayoutEdge(fromNode, toNode);
388 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
389 }
390
391 AddNeighbor(fromNode, toNode);
392 AddNeighbor(toNode, fromNode);
393 AddNode(fromNode);
394 AddNode(toNode);
395 return edge;
396 }
