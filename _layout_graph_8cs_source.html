132 private void ApplyVariations(RandomSeed randomSeed)
133 {
134 foreach (var pair in NodeVariations.OrderBy(x => x.Key))
135 {
136 var variations = randomSeed.Shuffled(pair.Value);
137
138 for (int i = 0; i < variations.Count; i++)
139 {
140 SwapEdges(pair.Value[i], variations[i]);
141 }
142 }

244 public void SwapEdges(int id1, int id2)
245 {
246 if (id1 == id2)
247 return;
248
249 // Get a set of all edges with the nodes.
250 var edges = GetEdges(id1).Concat(GetEdges(id2)).Distinct().ToList();
251
252 // Remove existing edges
253 foreach (var edge in edges)
254 {
255 RemoveEdge(edge.FromNode, edge.ToNode);
256 }
257
258 // Create new edges with the node replacements and copy properties to new edges.
259 foreach (var edge in edges)
260 {
261 var node1 = edge.FromNode == id1 ? id2 : edge.FromNode == id2 ? id1 : edge.FromNode;
262 var node2 = edge.ToNode == id1 ? id2 : edge.ToNode == id2 ? id1 : edge.ToNode;
263 AddEdge(node1, node2).SetProperties(edge);
264 }
265 }

295 public LayoutEdge AddEdge(int fromNode, int toNode)
296 {
297 if (fromNode == toNode)
298 throw new InvalidIdException($"From node cannot be the same as to node: {fromNode}.");
299
300 if (!TryGetEdge(fromNode, toNode, out var edge))
301 {
302 edge = new LayoutEdge(fromNode, toNode);
303 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
304 }
305
306 AddNode(fromNode);
307 AddNode(toNode);
308
309 var fromNeighbors = Neighbors[fromNode];
310 var toNeighbors = Neighbors[toNode];
311
312 if (!fromNeighbors.Contains(toNode))
313 fromNeighbors.Add(toNode);
314
315 if (!toNeighbors.Contains(fromNode))
316 toNeighbors.Add(fromNode);
317
318 return edge;
319 }

325 public void RemoveNode(int id)
326 {
327 var neighbors = Neighbors[id];
328
329 for (int i = neighbors.Count - 1; i >= 0; i--)
330 {
331 var neighbor = neighbors[i];
332 Neighbors[neighbor].Remove(id);
333
334 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
335 Edges.Remove(new EdgeIndexes(neighbor, id));
336 }
337
338 Neighbors.Remove(id);
339 Nodes.Remove(id);
340 RemoveNodeVariations(id);
341 }
