110 public LayoutEdge AddEdge(int fromNode, int toNode)
111 {
112 if (!TryGetEdge(fromNode, toNode, out var edge))
113 {
114 edge = new LayoutEdge(fromNode, toNode);
115 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
116 }
117
118 AddNode(fromNode);
119 AddNode(toNode);
120
121 var fromNeighbors = Neighbors[fromNode];
122 var toNeighbors = Neighbors[toNode];
123
124 if (!fromNeighbors.Contains(toNode))
125 fromNeighbors.Add(toNode);
126
127 if (!toNeighbors.Contains(fromNode))
128 toNeighbors.Add(fromNode);
129
130 return edge;
131 }

137 public void RemoveNode(int id)
138 {
139 var neighbors = Neighbors[id];
140
141 for (int i = neighbors.Count - 1; i >= 0; i--)
142 {
143 var neighbor = neighbors[i];
144 Neighbors[neighbor].Remove(id);
145
146 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
147 Edges.Remove(new EdgeIndexes(neighbor, id));
148 }
149
150 Neighbors.Remove(id);
151 Nodes.Remove(id);
152 }
