173 private void ApplyVariations(RandomSeed randomSeed)
174 {
175 foreach (var pair in NodeVariations.OrderBy(x => x.Key))
176 {
177 var variations = randomSeed.Shuffled(pair.Value);
178
179 for (int i = 0; i < variations.Count; i++)
180 {
181 SwapEdges(pair.Value[i], variations[i]);
182 }
183 }
184 }

285 public void SwapEdges(int id1, int id2)
286 {
287 if (id1 == id2)
288 return;
289
290 // Get a set of all edges with the nodes.
291 var edges = GetEdges(id1).Concat(GetEdges(id2)).Distinct().ToList();
292
293 // Remove existing edges
294 foreach (var edge in edges)
295 {
296 RemoveEdge(edge.FromNode, edge.ToNode);
297 }
298
299 // Create new edges with the node replacements and copy properties to new edges.
300 foreach (var edge in edges)
301 {
302 var node1 = edge.FromNode == id1 ? id2 : edge.FromNode == id2 ? id1 : edge.FromNode;
303 var node2 = edge.ToNode == id1 ? id2 : edge.ToNode == id2 ? id1 : edge.ToNode;
304 AddEdge(node1, node2).SetProperties(edge);
305 }
306 }

336 public LayoutEdge AddEdge(int fromNode, int toNode)
337 {
338 if (fromNode == toNode)
339 throw new InvalidIdException($"From node cannot be the same as to node: {fromNode}.");
340
341 if (!TryGetEdge(fromNode, toNode, out var edge))
342 {
343 edge = new LayoutEdge(fromNode, toNode);
344 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
345 }
346
347 AddNode(fromNode);
348 AddNode(toNode);
349
350 var fromNeighbors = Neighbors[fromNode];
351 var toNeighbors = Neighbors[toNode];
352
353 if (!fromNeighbors.Contains(toNode))
354 fromNeighbors.Add(toNode);
355
356 if (!toNeighbors.Contains(fromNode))
357 toNeighbors.Add(fromNode);
358
359 return edge;
360 }

366 public void RemoveNode(int id)
367 {
368 var neighbors = Neighbors[id];
369
370 for (int i = neighbors.Count - 1; i >= 0; i--)
371 {
372 var neighbor = neighbors[i];
373 Neighbors[neighbor].Remove(id);
374
375 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
376 Edges.Remove(new EdgeIndexes(neighbor, id));
377 }
378
379 Neighbors.Remove(id);
380 Nodes.Remove(id);
381 RemoveNodeVariations(id);
382 }

390 public void RemoveEdge(int node1, int node2)
391 {
392 Neighbors[node1].Remove(node2);
393 Neighbors[node2].Remove(node1);
394
395 if (!Edges.Remove(new EdgeIndexes(node1, node2)))
396 Edges.Remove(new EdgeIndexes(node2, node1));
