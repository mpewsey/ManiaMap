172 private void ApplyVariations(RandomSeed randomSeed)
173 {
174 foreach (var pair in NodeVariations.OrderBy(x => x.Key))
175 {
176 var variations = randomSeed.Shuffled(pair.Value);
177
178 for (int i = 0; i < variations.Count; i++)
179 {
180 SwapEdges(pair.Value[i], variations[i]);
181 }
182 }
183 }

284 public void SwapEdges(int id1, int id2)
285 {
286 if (id1 == id2)
287 return;
288
289 // Get a set of all edges with the nodes.
290 var edges = GetEdges(id1).Concat(GetEdges(id2)).Distinct().ToList();
291
292 // Remove existing edges
293 foreach (var edge in edges)
294 {
295 RemoveEdge(edge.FromNode, edge.ToNode);
296 }
297
298 // Create new edges with the node replacements and copy properties to new edges.
299 foreach (var edge in edges)
300 {
301 var node1 = edge.FromNode == id1 ? id2 : edge.FromNode == id2 ? id1 : edge.FromNode;
302 var node2 = edge.ToNode == id1 ? id2 : edge.ToNode == id2 ? id1 : edge.ToNode;
303 AddEdge(node1, node2).SetProperties(edge);
304 }
305 }

335 public LayoutEdge AddEdge(int fromNode, int toNode)
336 {
337 if (fromNode == toNode)
338 throw new InvalidIdException($"From node cannot be the same as to node: {fromNode}.");
339
340 if (!TryGetEdge(fromNode, toNode, out var edge))
341 {
342 edge = new LayoutEdge(fromNode, toNode);
343 Edges.Add(new EdgeIndexes(fromNode, toNode), edge);
344 }
345
346 AddNode(fromNode);
347 AddNode(toNode);
348
349 var fromNeighbors = Neighbors[fromNode];
350 var toNeighbors = Neighbors[toNode];
351
352 if (!fromNeighbors.Contains(toNode))
353 fromNeighbors.Add(toNode);
354
355 if (!toNeighbors.Contains(fromNode))
356 toNeighbors.Add(fromNode);
357
358 return edge;
359 }

365 public void RemoveNode(int id)
366 {
367 var neighbors = Neighbors[id];
368
369 for (int i = neighbors.Count - 1; i >= 0; i--)
370 {
371 var neighbor = neighbors[i];
372 Neighbors[neighbor].Remove(id);
373
374 if (!Edges.Remove(new EdgeIndexes(id, neighbor)))
375 Edges.Remove(new EdgeIndexes(neighbor, id));
376 }
377
378 Neighbors.Remove(id);
379 Nodes.Remove(id);
380 RemoveNodeVariations(id);
381 }

389 public void RemoveEdge(int node1, int node2)
390 {
391 Neighbors[node1].Remove(node2);
392 Neighbors[node2].Remove(node1);
393
394 if (!Edges.Remove(new EdgeIndexes(node1, node2)))
395 Edges.Remove(new EdgeIndexes(node2, node1));
396 }
