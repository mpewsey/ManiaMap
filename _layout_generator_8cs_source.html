99 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups, RandomSeed randomSeed)
100 {
101 Graph = graph;
102 TemplateGroups = templateGroups;
103 RandomSeed = randomSeed;
104 ConfigurationSpaces = templateGroups.GetConfigurationSpaces();
105
106 int chain = 0;
107 var chains = Graph.FindChains(MaxBranchLength);
108 var layouts = new Stack<Layout>();
109 layouts.Push(new Layout(layoutId, graph.Name, randomSeed));
110
111 while (layouts.Count > 0)
112 {
113 Layout = layouts.Peek();
114
115 // If all chains have been added, return the layout.
116 if (chain >= chains.Count)
117 return Layout;
118
119 // If layout has been used as a base more than the maximum allowable count,
120 // remove the layout and backtrack.
121 if (Layout.Rebases >= MaxRebases)
122 {
123 layouts.Pop();
124 chain--;
125 continue;
126 }
127
128 // Create a new layout and try to add the next chain.
129 Layout = new Layout(Layout);
130
131 if (AddChain(chains[chain]))
132 {
133 layouts.Push(Layout);
134 chain++;
135 }
136 }
137
138 return null;
139 }

167 private bool AddChain(List<LayoutEdge> chain)
168 {
169 for (int i = 0; i < chain.Count; i++)
170 {
171 var backEdge = chain[i];
172 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
173
174 if (CanInsertRoom(backEdge, aheadEdge))
175 {
176 if (!InsertRooms(backEdge, aheadEdge))
177 return false;
178 i++;
179 }
180 else if (!AddRooms(backEdge))
181 {
182 return false;
183 }
184 }
185
186 return true;
187 }

194 private bool CanInsertRoom(LayoutEdge backEdge, LayoutEdge aheadEdge)
195 {
196 return aheadEdge != null
197 && backEdge.ToNode == aheadEdge.FromNode
198 && Layout.Rooms.ContainsKey(new Uid(aheadEdge.ToNode))
199 && Layout.Rooms.ContainsKey(new Uid(backEdge.FromNode))
200 && !Layout.Rooms.ContainsKey(new Uid(backEdge.ToNode));

208 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
209 {
210 var midNode = Graph.GetNode(backEdge.ToNode);
211 var backNode = Graph.GetNode(backEdge.FromNode);
212 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
213
214 // Try to add edge rooms.
215 var addBackEdgeRoom = backEdge.RoomChanceSatisfied(RandomSeed.Random.NextDouble());
216 var addAheadEdgeRoom = aheadEdge.RoomChanceSatisfied(RandomSeed.Random.NextDouble());
217 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
218 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
219
220 // Try to insert the node room.
221 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
222 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
223
224 return InsertRoom(midNode,
225 backRoomId, backEdge.DoorCode, backEdge.Direction,
226 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
227 }

234 private bool AddRooms(LayoutEdge edge)
235 {
236 var fromNode = Graph.GetNode(edge.FromNode);
237 var toNode = Graph.GetNode(edge.ToNode);
238 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
239 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
240 var addEdgeRoom = edge.RoomChanceSatisfied(RandomSeed.Random.NextDouble());
241
242 if (toRoomExists)
243 throw new Exception("To Node Exists. Chains are not properly ordered.");
244
245 // If both rooms do not exist, add the first room. If that fails, abort.
246 if (!fromRoomExists && !AddFirstRoom(fromNode))
247 return false;
248
249 // Try to add a room for the edge.
250 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
251
252 // Try to add a room for the to node.
253 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
254 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
255 }

281 private bool AddRoom(IRoomSource source,
282 Uid fromRoomId, int code, EdgeDirection direction)
283 {
284 var fromRoom = Layout.Rooms[fromRoomId];
285 var z = source.Z - fromRoom.Position.Z;
286
287 foreach (var template in GetTemplates(source.TemplateGroup))
288 {
289 foreach (var config in GetConfigurations(fromRoom.Template, template))
290 {
291 var position = config.Position + fromRoom.Position;
292
293 // Check if the configuration can be added to the layout. If not, try the next one.
294 if (!config.Matches(z, code, direction))
295 continue;
296 if (Layout.Intersects(template, position, source.Z))
297 continue;
298
299 // Add the room to the layout.
300 var room = new Room(source, position, template, RandomSeed);
301 Layout.Rooms.Add(room.Id, room);
302
303 // Try to create a door connection between the rooms.
304 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
305 {
306 // Remove the room added previously.
307 Layout.Rooms.Remove(room.Id);
308 continue;
309 }
310
311 return true;
312 }
313 }
314
315 return false;
316 }

328 private bool InsertRoom(IRoomSource source,
329 Uid backRoomId, int backCode, EdgeDirection backDirection,
330 Uid aheadRoomId, int aheadCode, EdgeDirection aheadDirection)
331 {
332 var backRoom = Layout.Rooms[backRoomId];
333 var aheadRoom = Layout.Rooms[aheadRoomId];
334 var z1 = source.Z - backRoom.Position.Z;
335 var z2 = aheadRoom.Position.Z - source.Z;
336
337 foreach (var template in GetTemplates(source.TemplateGroup))
338 {
339 foreach (var config1 in GetConfigurations(backRoom.Template, template))
340 {
341 var position = config1.Position + backRoom.Position;
342
343 // Check if the configuration can be added to the layout. If not, try the next one.
344 if (!config1.Matches(z1, backCode, backDirection))
345 continue;
346 if (Layout.Intersects(template, position, source.Z))
347 continue;
348
349 foreach (var config2 in GetConfigurations(template, aheadRoom.Template))
350 {
351 var offset = aheadRoom.Position - position;
352
353 // Check if the configuration can be added to the layout. If not, try the next one.
354 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
355 continue;
356
357 var room = new Room(source, position, template, RandomSeed);
358 Layout.Rooms.Add(room.Id, room);
359
360 // Try to create a door connection between the rooms.
361 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
362 {
363 // Remove the room added previously.
364 Layout.Rooms.Remove(room.Id);
365 continue;
366 }
367
368 // Try to create a door connection between the rooms.
369 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
370 {
371 // Remove the room and door connection added previously.
372 Layout.Rooms.Remove(room.Id);
373 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
374 continue;
375 }
376
377 return true;
378 }
379 }
380 }
381
382 return false;

392 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
393 {
394 var fromRoom = Layout.Rooms[fromRoomId];
395 var toRoom = Layout.Rooms[toRoomId];
396 var fromDoor = config.FromDoor;
397 var toDoor = config.ToDoor;
398
399 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) <= 1)
400 {
401 // Rooms do not require shaft connection. Simply add the door connection.
402 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor);
403 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
404 }
405 else
406 {
407 // Shaft is required.
408 var position = fromDoor.Position + fromRoom.Position;
409 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
410 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
411 var min = new Vector3DInt(position.X, position.Y, zMin);
412 var max = new Vector3DInt(position.X, position.Y, zMax);
413
414 // If shaft intersects the layout, abort adding the door connection.
415 if (Layout.Intersects(min, max))
416 return false;
417
418 var shaft = new Box(min, max);
419 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
420 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
421 }
422
423 return true;
424 }
