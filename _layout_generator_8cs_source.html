78 public Layout GenerateLayout(int id, RandomSeed random)
79 {
80 RandomSeed = random;
81 ConfigurationSpaces = TemplateGroups.GetConfigurationSpaces();
82
83 int chain = 0;
84 var chains = Graph.FindChains(MaxBranchLength);
85 var layouts = new Stack<Layout>();
86 layouts.Push(new Layout(id, Graph.Name, RandomSeed));
87
88 while (layouts.Count > 0)
89 {
90 Layout = layouts.Peek();
91
92 // If all chains have been added, return the layout.
93 if (chain >= chains.Count)
94 return Layout;
95
96 // If layout has been used as a base more than the maximum allowable count,
97 // remove the layout and backtrack.
98 if (Layout.Rebases >= MaxRebases)
99 {
100 layouts.Pop();
101 chain--;
102 continue;
103 }
104
105 // Create a new layout and try to add the next chain.
106 Layout = new Layout(Layout);
107
108 if (AddChain(chains[chain]))
109 {
110 layouts.Push(Layout);
111 chain++;
112 }
113 }
114
115 return null;
116 }

146 private bool AddChain(List<LayoutEdge> chain)
147 {
148 for (int i = 0; i < chain.Count; i++)
149 {
150 var backEdge = chain[i];
151 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
152
153 if (CanInsertRoom(backEdge, aheadEdge))
154 {
155 if (!InsertRooms(backEdge, aheadEdge))
156 return false;
157 i++;
158 }
159 else if (!AddRooms(backEdge))
160 {
161 return false;
162 }
163 }
164
165 return true;
166 }

173 private bool CanInsertRoom(LayoutEdge backEdge, LayoutEdge aheadEdge)
174 {
175 return aheadEdge != null
176 && backEdge.ToNode == aheadEdge.FromNode
177 && Layout.Rooms.ContainsKey(new Uid(aheadEdge.ToNode))
178 && Layout.Rooms.ContainsKey(new Uid(backEdge.FromNode))
179 && !Layout.Rooms.ContainsKey(new Uid(backEdge.ToNode));

187 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
188 {
189 var midNode = Graph.GetNode(backEdge.ToNode);
190 var backNode = Graph.GetNode(backEdge.FromNode);
191 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
192
193 // Try to add edge rooms.
194 var addBackEdgeRoom = backEdge.RoomChanceSatisfied(RandomSeed.Random.NextDouble());
195 var addAheadEdgeRoom = aheadEdge.RoomChanceSatisfied(RandomSeed.Random.NextDouble());
196 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
197 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
198
199 // Try to insert the node room.
200 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
201 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
202
203 return InsertRoom(midNode,
204 backRoomId, backEdge.DoorCode, backEdge.Direction,
205 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
206 }

213 private bool AddRooms(LayoutEdge edge)
214 {
215 var fromNode = Graph.GetNode(edge.FromNode);
216 var toNode = Graph.GetNode(edge.ToNode);
217 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
218 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
219 var addEdgeRoom = edge.RoomChanceSatisfied(RandomSeed.Random.NextDouble());
220
221 if (toRoomExists)
222 throw new Exception("To Node Exists. Chains are not properly ordered.");
223
224 // If both rooms do not exist, add the first room. If that fails, abort.
225 if (!fromRoomExists && !AddFirstRoom(fromNode))
226 return false;
227
228 // Try to add a room for the edge.
229 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
230
231 // Try to add a room for the to node.
232 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
233 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
234 }

260 private bool AddRoom(IRoomSource source,
261 Uid fromRoomId, int code, EdgeDirection direction)
262 {
263 var fromRoom = Layout.Rooms[fromRoomId];
264 var z = source.Z - fromRoom.Position.Z;
265
266 foreach (var template in GetTemplates(source.TemplateGroup))
267 {
268 foreach (var config in GetConfigurations(fromRoom.Template, template))
269 {
270 var position = config.Position + fromRoom.Position;
271
272 // Check if the configuration can be added to the layout. If not, try the next one.
273 if (!config.Matches(z, code, direction))
274 continue;
275 if (Layout.Intersects(template, position, source.Z))
276 continue;
277
278 // Add the room to the layout.
279 var room = new Room(source, position, template, RandomSeed);
280 Layout.Rooms.Add(room.Id, room);
281
282 // Try to create a door connection between the rooms.
283 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
284 {
285 // Remove the room added previously.
286 Layout.Rooms.Remove(room.Id);
287 continue;
288 }
289
290 return true;
291 }
292 }
293
294 return false;
295 }

307 private bool InsertRoom(IRoomSource source,
308 Uid backRoomId, int backCode, EdgeDirection backDirection,
309 Uid aheadRoomId, int aheadCode, EdgeDirection aheadDirection)
310 {
311 var backRoom = Layout.Rooms[backRoomId];
312 var aheadRoom = Layout.Rooms[aheadRoomId];
313 var z1 = source.Z - backRoom.Position.Z;
314 var z2 = aheadRoom.Position.Z - source.Z;
315
316 foreach (var template in GetTemplates(source.TemplateGroup))
317 {
318 foreach (var config1 in GetConfigurations(backRoom.Template, template))
319 {
320 var position = config1.Position + backRoom.Position;
321
322 // Check if the configuration can be added to the layout. If not, try the next one.
323 if (!config1.Matches(z1, backCode, backDirection))
324 continue;
325 if (Layout.Intersects(template, position, source.Z))
326 continue;
327
328 foreach (var config2 in GetConfigurations(template, aheadRoom.Template))
329 {
330 var offset = aheadRoom.Position - position;
331
332 // Check if the configuration can be added to the layout. If not, try the next one.
333 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
334 continue;
335
336 var room = new Room(source, position, template, RandomSeed);
337 Layout.Rooms.Add(room.Id, room);
338
339 // Try to create a door connection between the rooms.
340 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
341 {
342 // Remove the room added previously.
343 Layout.Rooms.Remove(room.Id);
344 continue;
345 }
346
347 // Try to create a door connection between the rooms.
348 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
349 {
350 // Remove the room and door connection added previously.
351 Layout.Rooms.Remove(room.Id);
352 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
353 continue;
354 }
355
356 return true;
357 }
358 }
359 }

361 return false;

371 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
372 {
373 var fromRoom = Layout.Rooms[fromRoomId];
374 var toRoom = Layout.Rooms[toRoomId];
375 var fromDoor = config.FromDoor;
376 var toDoor = config.ToDoor;
377
378 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) <= 1)
379 {
380 // Rooms do not require shaft connection. Simply add the door connection.
381 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor);
382 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
383 }
384 else
385 {
386 // Shaft is required.
387 var position = fromDoor.Position + fromRoom.Position;
388 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
389 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
390 var min = new Vector3DInt(position.X, position.Y, zMin);
391 var max = new Vector3DInt(position.X, position.Y, zMax);
392
393 // If shaft intersects the layout, abort adding the door connection.
394 if (Layout.Intersects(min, max))
395 return false;
396
397 var shaft = new Box(min, max);
398 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
399 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
400 }
401
402 return true;
403 }
