111 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups, RandomSeed randomSeed)
112 {
113 Graph = graph;
114 TemplateGroups = templateGroups;
115 RandomSeed = randomSeed;
116 Validate();
117 ConfigurationSpaces = templateGroups.GetConfigurationSpaces();
118
119 int chain = 0;
120 var allowableRebases = AllowableRebases(chain);
121 var chains = Graph.FindChains(MaxBranchLength);
122 var layouts = new Stack<Layout>();
123 layouts.Push(new Layout(layoutId, graph.Name, randomSeed));
124
125 while (layouts.Count > 0)
126 {
127 Layout = layouts.Peek();
128
129 // If all chains have been added, return the layout.
130 if (chain >= chains.Count)
131 return Layout;
132
133 // If layout has been used as a base more than the maximum allowable count,
134 // remove the layout and backtrack.
135 if (Layout.Rebases > allowableRebases)
136 {
137 layouts.Pop();
138 chain--;
139 allowableRebases = AllowableRebases(chain);
140 continue;
141 }
142
143 // Create a new layout and try to add the next chain.
144 Layout = new Layout(Layout);
145
146 if (AddChain(chains[chain]))
147 {
148 layouts.Push(Layout);
149 chain++;
150 allowableRebases = AllowableRebases(chain);
151 }
152 }
153
154 return null;
155 }

202 private bool AddChain(List<LayoutEdge> chain)
203 {
204 for (int i = 0; i < chain.Count; i++)
205 {
206 var backEdge = chain[i];
207 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
208
209 if (CanInsertRoom(backEdge, aheadEdge))
210 {
211 if (!InsertRooms(backEdge, aheadEdge))
212 return false;
213 i++;
214 }
215 else if (!AddRooms(backEdge))
216 {
217 return false;
218 }
219 }
220
221 return true;
222 }

243 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
244 {
245 var midNode = Graph.GetNode(backEdge.ToNode);
246 var backNode = Graph.GetNode(backEdge.FromNode);
247 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
248
249 // Try to add edge rooms.
250 var addBackEdgeRoom = backEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
251 var addAheadEdgeRoom = aheadEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
252 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
253 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
254
255 // Try to insert the node room.
256 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
257 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
258
259 return InsertRoom(midNode,
260 backRoomId, backEdge.DoorCode, backEdge.Direction,
261 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
262 }

269 private bool AddRooms(LayoutEdge edge)
270 {
271 var fromNode = Graph.GetNode(edge.FromNode);
272 var toNode = Graph.GetNode(edge.ToNode);
273 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
274 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
275 var addEdgeRoom = edge.RoomChanceSatisfied(RandomSeed.NextDouble());
276
277 if (toRoomExists)
278 throw new InvalidChainOrderException("To Node Exists. Chains are not properly ordered.");
279
280 // If both rooms do not exist, add the first room. If that fails, abort.
281 if (!fromRoomExists && !AddFirstRoom(fromNode))
282 return false;
283
284 // Try to add a room for the edge.
285 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
286
287 // Try to add a room for the to node.
288 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
289 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
290 }

316 private bool AddRoom(IRoomSource source,
317 Uid fromRoomId, int code, EdgeDirection direction)
318 {
319 var fromRoom = Layout.Rooms[fromRoomId];
320 var z = source.Z - fromRoom.Position.Z;
321
322 foreach (var template in GetTemplates(source.TemplateGroup))
323 {
324 foreach (var config in GetConfigurations(fromRoom.Template, template))
325 {
326 var position = config.Position + fromRoom.Position;
327
328 // Check if the configuration can be added to the layout. If not, try the next one.
329 if (!config.Matches(z, code, direction))
330 continue;
331 if (Layout.Intersects(template, position, source.Z))
332 continue;
333
334 // Add the room to the layout.
335 var room = new Room(source, position, template, RandomSeed);
336 Layout.Rooms.Add(room.Id, room);
337
338 // Try to create a door connection between the rooms.
339 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
340 {
341 // Remove the room added previously.
342 Layout.Rooms.Remove(room.Id);
343 continue;
344 }
345
346 return true;
347 }
348 }
349
350 return false;
351 }

363 private bool InsertRoom(IRoomSource source,
364 Uid backRoomId, int backCode, EdgeDirection backDirection,
365 Uid aheadRoomId, int aheadCode, EdgeDirection aheadDirection)
366 {
367 var backRoom = Layout.Rooms[backRoomId];
368 var aheadRoom = Layout.Rooms[aheadRoomId];
369 var z1 = source.Z - backRoom.Position.Z;
370 var z2 = aheadRoom.Position.Z - source.Z;
371
372 foreach (var template in GetTemplates(source.TemplateGroup))
373 {
374 foreach (var config1 in GetConfigurations(backRoom.Template, template))
375 {
376 var position = config1.Position + backRoom.Position;
377
378 // Check if the configuration can be added to the layout. If not, try the next one.
379 if (!config1.Matches(z1, backCode, backDirection))
380 continue;
381 if (Layout.Intersects(template, position, source.Z))
382 continue;
383
384 foreach (var config2 in GetConfigurations(template, aheadRoom.Template))
385 {
386 var offset = aheadRoom.Position - position;
387
388 // Check if the configuration can be added to the layout. If not, try the next one.
389 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
390 continue;
391
392 var room = new Room(source, position, template, RandomSeed);
393 Layout.Rooms.Add(room.Id, room);
394
395 // Try to create a door connection between the rooms.
396 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
397 {
398 // Remove the room added previously.
399 Layout.Rooms.Remove(room.Id);
400 continue;
401 }
402
403 // Try to create a door connection between the rooms.
404 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
405 {
406 // Remove the room and door connection added previously.
407 Layout.Rooms.Remove(room.Id);
408 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
409 continue;
410 }
411
412 return true;
413 }
414 }
415 }
416
417 return false;

427 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
428 {
429 var fromRoom = Layout.Rooms[fromRoomId];
430 var toRoom = Layout.Rooms[toRoomId];
431 var fromDoor = config.FromDoor;
432 var toDoor = config.ToDoor;
433
434 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) <= 1)
435 {
436 // Rooms do not require shaft connection. Simply add the door connection.
437 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor);
438 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
439 }
440 else
441 {
442 // Shaft is required.
443 var position = fromDoor.Position + fromRoom.Position;
444 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
445 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
446 var min = new Vector3DInt(position.X, position.Y, zMin);
447 var max = new Vector3DInt(position.X, position.Y, zMax);
448
449 // If shaft intersects the layout, abort adding the door connection.
450 if (Layout.Intersects(min, max))
451 return false;
452
453 var shaft = new Box(min, max);
454 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
455 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
456 }
457
458 return true;
459 }
