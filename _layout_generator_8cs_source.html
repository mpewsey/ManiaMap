159 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups, RandomSeed randomSeed)
160 {
161 GenerationLogger.Log("Running layout generator...");
162 Initialize(graph, templateGroups, randomSeed);
163
164 var chains = Graph.FindChains(MaxBranchLength);
165 var baseLayout = new Layout(layoutId, graph.Name, randomSeed);
166 var layouts = new Stack<Layout>(chains.Count);
167 layouts.Push(baseLayout);
168
169 while (layouts.Count > 0)
170 {
171 Layout = layouts.Peek();
172
173 // If all chains have been added, validate the layout.
174 if (ChainIndex >= chains.Count)
175 {
176 // If layout is complete, return the layout.
177 if (Layout.IsComplete(TemplateGroups))
178 {
179 Layout = new Layout(Layout);
180 GenerationLogger.Log("Layout generator complete.");
181 return Layout;
182 }
183
184 // If layout is not complete, start over.
185 ChainIndex = 0;
186 layouts.Clear();
187 layouts.Push(baseLayout);
188 GenerationLogger.Log("Layout constraints not satisfied. Restarting...");
189 continue;
190 }
191
192 // If layout has been used as a base more than the maximum allowable count,
193 // remove the layout and backtrack.
194 if (Layout.Rebases > AllowableRebases)
195 {
196 layouts.Pop();
197 ChainIndex--;
198 GenerationLogger.Log("Rebase count exceeded. Backtracking...");
199 continue;
200 }
201
202 // Create a new layout and try to add the next chain.
203 Layout = new Layout(Layout);
204
205 if (AddChain(chains[ChainIndex]))
206 {
207 layouts.Push(Layout);
208 ChainIndex++;
209 GenerationLogger.Log($"Added chain {ChainIndex} / {chains.Count}...");
210 }
211 }
212
213 GenerationLogger.Log("Layout generator failed.");
214 return null;
215 }

253 private bool AddChain(List<LayoutEdge> chain)
254 {
255 for (int i = 0; i < chain.Count; i++)
256 {
257 var backEdge = chain[i];
258 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
259
260 if (CanInsertRoom(backEdge, aheadEdge))
261 {
262 if (!InsertRooms(backEdge, aheadEdge))
263 return false;
264 i++;
265 }
266 else if (!AddRooms(backEdge))
267 {
268 return false;
269 }
270 }
271
272 return true;

294 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
295 {
296 var midNode = Graph.GetNode(backEdge.ToNode);
297 var backNode = Graph.GetNode(backEdge.FromNode);
298 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
299
300 // Try to add edge rooms.
301 var addBackEdgeRoom = backEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
302 var addAheadEdgeRoom = aheadEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
303 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
304 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
305
306 // If a back edge room is required and was not added, abort.
307 if (!addedBackEdgeRoom && addBackEdgeRoom && backEdge.RequireRoom)
308 return false;
309
310 // If an ahead edge room is required and was not added, abort.
311 if (!addedAheadEdgeRoom && addAheadEdgeRoom && aheadEdge.RequireRoom)
312 return false;
313
314 // Try to insert the node room.
315 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
316 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
317
318 return InsertRoom(midNode,
319 backRoomId, backEdge.DoorCode, backEdge.Direction,
320 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
321 }

328 private bool AddRooms(LayoutEdge edge)
329 {
330 var fromNode = Graph.GetNode(edge.FromNode);
331 var toNode = Graph.GetNode(edge.ToNode);
332 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
333 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
334 var addEdgeRoom = edge.RoomChanceSatisfied(RandomSeed.NextDouble());
335
336 if (toRoomExists)
337 throw new InvalidChainOrderException("To Node Exists. Chains are not properly ordered.");
338
339 // If both rooms do not exist, add the first room. If that fails, abort.
340 if (!fromRoomExists && !AddFirstRoom(fromNode))
341 return false;
342
343 // Try to add a room for the edge.
344 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
345
346 // If an edge room is required and was not added, abort.
347 if (!addedEdgeRoom && addEdgeRoom && edge.RequireRoom)
348 return false;
349
350 // Try to add a room for the to node.
351 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
352 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
353 }

380 private bool AddRoom(IRoomSource source,
381 Uid fromRoomId, int code, EdgeDirection direction)
382 {
383 var fromRoom = Layout.Rooms[fromRoomId];
384 var z = source.Z - fromRoom.Position.Z;
385
386 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
387 {
388 foreach (var config in GetConfigurations(fromRoom.Template, entry.Template))
389 {
390 var position = config.Position + fromRoom.Position;
391
392 // Check if the configuration can be added to the layout. If not, try the next one.
393 if (!config.Matches(z, code, direction))
394 continue;
395 if (Layout.Intersects(entry.Template, position, source.Z))
396 continue;
397
398 // Add the room to the layout.
399 var room = new Room(source, position, entry.Template, RandomSeed);
400 Layout.Rooms.Add(room.Id, room);
401
402 // Try to create a door connection between the rooms.
403 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
404 {
405 // Remove the room added previously.
406 Layout.Rooms.Remove(room.Id);
407 continue;
408 }
409
410 Layout.IncreaseTemplateCount(entry);
411 return true;
412 }
413 }
414
415 return false;

428 private bool InsertRoom(IRoomSource source,
429 Uid backRoomId, int backCode, EdgeDirection backDirection,
430 Uid aheadRoomId, int aheadCode, EdgeDirection aheadDirection)
431 {
432 var backRoom = Layout.Rooms[backRoomId];
433 var aheadRoom = Layout.Rooms[aheadRoomId];
434 var z1 = source.Z - backRoom.Position.Z;
435 var z2 = aheadRoom.Position.Z - source.Z;
436
437 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
438 {
439 foreach (var config1 in GetConfigurations(backRoom.Template, entry.Template))
440 {
441 var position = config1.Position + backRoom.Position;
442
443 // Check if the configuration can be added to the layout. If not, try the next one.
444 if (!config1.Matches(z1, backCode, backDirection))
445 continue;
446 if (Layout.Intersects(entry.Template, position, source.Z))
447 continue;
448
449 foreach (var config2 in GetConfigurations(entry.Template, aheadRoom.Template))
450 {
451 var offset = aheadRoom.Position - position;
452
453 // Check if the configuration can be added to the layout. If not, try the next one.
454 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
455 continue;
456
457 var room = new Room(source, position, entry.Template, RandomSeed);
458 Layout.Rooms.Add(room.Id, room);
459
460 // Try to create a door connection between the rooms.
461 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
462 {
463 // Remove the room added previously.
464 Layout.Rooms.Remove(room.Id);
465 continue;
466 }
467
468 // Try to create a door connection between the rooms.
469 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
470 {
471 // Remove the room and door connection added previously.
472 Layout.Rooms.Remove(room.Id);
473 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
474 continue;
475 }
476
477 Layout.IncreaseTemplateCount(entry);
478 return true;
479 }
480 }
481 }
482
483 return false;

493 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
494 {
495 var fromRoom = Layout.Rooms[fromRoomId];
496 var toRoom = Layout.Rooms[toRoomId];
497 var fromDoor = config.FromDoor;
498 var toDoor = config.ToDoor;
499 Box shaft = null;
500
501 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) > 1)
502 {
503 // Shaft is required.
504 var position = fromDoor.Position + fromRoom.Position;
505 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
506 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
507 var min = new Vector3DInt(position.X, position.Y, zMin);
508 var max = new Vector3DInt(position.X, position.Y, zMax);
509
510 // If shaft intersects the layout, abort adding the door connection.
511 if (Layout.Intersects(min, max))
512 return false;
513
514 shaft = new Box(min, max);
515 }
516
517 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
518 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
519 return true;
520 }
