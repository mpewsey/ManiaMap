126 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups, RandomSeed randomSeed)
127 {
128 Initialize(graph, templateGroups, randomSeed);
129
130 int chain = 0;
131 var allowableRebases = AllowableRebases(chain);
132 var chains = Graph.FindChains(MaxBranchLength);
133 var layouts = new Stack<Layout>(chains.Count);
134 layouts.Push(new Layout(layoutId, graph.Name, randomSeed));
135
136 while (layouts.Count > 0)
137 {
138 Layout = layouts.Peek();
139
140 // If all chains have been added, return the layout.
141 if (chain >= chains.Count)
142 return new Layout(Layout);
143
144 // If layout has been used as a base more than the maximum allowable count,
145 // remove the layout and backtrack.
146 if (Layout.Rebases > allowableRebases)
147 {
148 layouts.Pop();
149 allowableRebases = AllowableRebases(--chain);
150 continue;
151 }
152
153 // Create a new layout and try to add the next chain.
154 Layout = new Layout(Layout);
155
156 if (AddChain(chains[chain]))
157 {
158 layouts.Push(Layout);
159 allowableRebases = AllowableRebases(++chain);
160 }
161 }
162
163 return null;
164 }

211 private bool AddChain(List<LayoutEdge> chain)
212 {
213 for (int i = 0; i < chain.Count; i++)
214 {
215 var backEdge = chain[i];
216 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
217
218 if (CanInsertRoom(backEdge, aheadEdge))
219 {
220 if (!InsertRooms(backEdge, aheadEdge))
221 return false;
222 i++;
223 }
224 else if (!AddRooms(backEdge))
225 {
226 return false;
227 }
228 }
229
230 return true;
231 }

252 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
253 {
254 var midNode = Graph.GetNode(backEdge.ToNode);
255 var backNode = Graph.GetNode(backEdge.FromNode);
256 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
257
258 // Try to add edge rooms.
259 var addBackEdgeRoom = backEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
260 var addAheadEdgeRoom = aheadEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
261 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
262 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
263
264 // If a back edge room is required and was not added, abort.
265 if (!addedBackEdgeRoom && addBackEdgeRoom && backEdge.RequireRoom)
266 return false;
267
268 // If an ahead edge room is required and was not added, abort.
269 if (!addedAheadEdgeRoom && addAheadEdgeRoom && aheadEdge.RequireRoom)
270 return false;
271
272 // Try to insert the node room.
273 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
274 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
275
276 return InsertRoom(midNode,
277 backRoomId, backEdge.DoorCode, backEdge.Direction,
278 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
279 }

286 private bool AddRooms(LayoutEdge edge)
287 {
288 var fromNode = Graph.GetNode(edge.FromNode);
289 var toNode = Graph.GetNode(edge.ToNode);
290 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
291 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
292 var addEdgeRoom = edge.RoomChanceSatisfied(RandomSeed.NextDouble());
293
294 if (toRoomExists)
295 throw new InvalidChainOrderException("To Node Exists. Chains are not properly ordered.");
296
297 // If both rooms do not exist, add the first room. If that fails, abort.
298 if (!fromRoomExists && !AddFirstRoom(fromNode))
299 return false;
300
301 // Try to add a room for the edge.
302 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
303
304 // If an edge room is required and was not added, abort.
305 if (!addedEdgeRoom && addEdgeRoom && edge.RequireRoom)
306 return false;
307
308 // Try to add a room for the to node.
309 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
310 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
311 }

337 private bool AddRoom(IRoomSource source,
338 Uid fromRoomId, int code, EdgeDirection direction)
339 {
340 var fromRoom = Layout.Rooms[fromRoomId];
341 var z = source.Z - fromRoom.Position.Z;
342
343 foreach (var template in GetTemplates(source.TemplateGroup))
344 {
345 foreach (var config in GetConfigurations(fromRoom.Template, template))
346 {
347 var position = config.Position + fromRoom.Position;
348
349 // Check if the configuration can be added to the layout. If not, try the next one.
350 if (!config.Matches(z, code, direction))
351 continue;
352 if (Layout.Intersects(template, position, source.Z))
353 continue;
354
355 // Add the room to the layout.
356 var room = new Room(source, position, template, RandomSeed);
357 Layout.Rooms.Add(room.Id, room);
358
359 // Try to create a door connection between the rooms.
360 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
361 {
362 // Remove the room added previously.
363 Layout.Rooms.Remove(room.Id);
364 continue;
365 }
366
367 return true;
368 }
369 }
370
371 return false;

384 private bool InsertRoom(IRoomSource source,
385 Uid backRoomId, int backCode, EdgeDirection backDirection,
386 Uid aheadRoomId, int aheadCode, EdgeDirection aheadDirection)
387 {
388 var backRoom = Layout.Rooms[backRoomId];
389 var aheadRoom = Layout.Rooms[aheadRoomId];
390 var z1 = source.Z - backRoom.Position.Z;
391 var z2 = aheadRoom.Position.Z - source.Z;
392
393 foreach (var template in GetTemplates(source.TemplateGroup))
394 {
395 foreach (var config1 in GetConfigurations(backRoom.Template, template))
396 {
397 var position = config1.Position + backRoom.Position;
398
399 // Check if the configuration can be added to the layout. If not, try the next one.
400 if (!config1.Matches(z1, backCode, backDirection))
401 continue;
402 if (Layout.Intersects(template, position, source.Z))
403 continue;
404
405 foreach (var config2 in GetConfigurations(template, aheadRoom.Template))
406 {
407 var offset = aheadRoom.Position - position;
408
409 // Check if the configuration can be added to the layout. If not, try the next one.
410 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
411 continue;
412
413 var room = new Room(source, position, template, RandomSeed);
414 Layout.Rooms.Add(room.Id, room);
415
416 // Try to create a door connection between the rooms.
417 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
418 {
419 // Remove the room added previously.
420 Layout.Rooms.Remove(room.Id);
421 continue;
422 }
423
424 // Try to create a door connection between the rooms.
425 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
426 {
427 // Remove the room and door connection added previously.
428 Layout.Rooms.Remove(room.Id);
429 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
430 continue;
431 }
432
433 return true;
434 }
435 }
436 }
437
438 return false;

448 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
449 {
450 var fromRoom = Layout.Rooms[fromRoomId];
451 var toRoom = Layout.Rooms[toRoomId];
452 var fromDoor = config.FromDoor;
453 var toDoor = config.ToDoor;
454
455 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) <= 1)
456 {
457 // Rooms do not require shaft connection. Simply add the door connection.
458 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor);
459 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
460 }
461 else
462 {
463 // Shaft is required.
464 var position = fromDoor.Position + fromRoom.Position;
465 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
466 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
467 var min = new Vector3DInt(position.X, position.Y, zMin);
468 var max = new Vector3DInt(position.X, position.Y, zMax);
469
470 // If shaft intersects the layout, abort adding the door connection.
471 if (Layout.Intersects(min, max))
472 return false;
473
474 var shaft = new Box(min, max);
475 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
476 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
477 }
478
479 return true;
480 }
