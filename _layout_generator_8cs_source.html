163 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups, RandomSeed randomSeed)
164 {
165 Logger.Log("[Layout Generator] Running layout generator...");
166 Initialize(graph, templateGroups, randomSeed);
167
168 var chains = Graph.FindChains(MaxBranchLength);
169 var baseLayout = new Layout(layoutId, graph.Name, randomSeed.Seed);
170 var layouts = new Stack<Layout>(chains.Count);
171 layouts.Push(baseLayout);
172
173 while (layouts.Count > 0)
174 {
175 Layout = layouts.Peek();
176
177 // If all chains have been added, validate the layout.
178 if (ChainIndex >= chains.Count)
179 {
180 // If layout is complete, return the layout.
181 if (Layout.IsComplete(TemplateGroups))
182 {
183 Layout = new Layout(Layout);
184 Logger.Log("[Layout Generator] Layout generator complete.");
185 return Layout;
186 }
187
188 // If layout is not complete, start over.
189 ChainIndex = 0;
190 layouts.Clear();
191 layouts.Push(baseLayout);
192 Logger.Log("[Layout Generator] Layout constraints not satisfied. Restarting...");
193 continue;
194 }
195
196 // If layout has been used as a base more than the maximum allowable count,
197 // remove the layout and backtrack.
198 if (Layout.Rebases > AllowableRebases)
199 {
200 layouts.Pop();
201 ChainIndex--;
202 Logger.Log("[Layout Generator] Rebase count exceeded. Backtracking...");
203 continue;
204 }
205
206 // Create a new layout and try to add the next chain.
207 Layout = new Layout(Layout);
208
209 if (AddChain(chains[ChainIndex]))
210 {
211 layouts.Push(Layout);
212 ChainIndex++;
213 Logger.Log($"[Layout Generator] Added chain {ChainIndex} / {chains.Count}...");
214 }
215 }
216
217 Logger.Log("[Layout Generator] Layout generator failed.");
218 return null;
219 }

257 private bool AddChain(List<LayoutEdge> chain)
258 {
259 for (int i = 0; i < chain.Count; i++)
260 {
261 var backEdge = chain[i];
262 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
263
264 if (CanInsertRoom(backEdge, aheadEdge))
265 {
266 if (!InsertRooms(backEdge, aheadEdge))
267 return false;
268 i++;
269 }
270 else if (!AddRooms(backEdge))
271 {
272 return false;
273 }
274 }
275
276 return true;
277 }

284 private bool CanInsertRoom(LayoutEdge backEdge, LayoutEdge aheadEdge)
285 {
286 return aheadEdge != null

298 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
299 {
300 var midNode = Graph.GetNode(backEdge.ToNode);
301 var backNode = Graph.GetNode(backEdge.FromNode);
302 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
303
304 // Try to add edge rooms.
305 var addBackEdgeRoom = RandomSeed.ChanceSatisfied(backEdge.RoomChance);
306 var addAheadEdgeRoom = RandomSeed.ChanceSatisfied(aheadEdge.RoomChance);
307 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
308 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
309
310 // If a back edge room is required and was not added, abort.
311 if (!addedBackEdgeRoom && addBackEdgeRoom && backEdge.RequireRoom)
312 return false;
313
314 // If an ahead edge room is required and was not added, abort.
315 if (!addedAheadEdgeRoom && addAheadEdgeRoom && aheadEdge.RequireRoom)
316 return false;
317
318 // Try to insert the node room.
319 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
320 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
321
322 return InsertRoom(midNode,
323 backRoomId, backEdge.DoorCode, backEdge.Direction,
324 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
325 }

332 private bool AddRooms(LayoutEdge edge)
333 {
334 var fromNode = Graph.GetNode(edge.FromNode);
335 var toNode = Graph.GetNode(edge.ToNode);
336 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
337 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
338 var addEdgeRoom = RandomSeed.ChanceSatisfied(edge.RoomChance);
339
340 if (toRoomExists)
341 throw new InvalidChainOrderException("To Node Exists. Chains are not properly ordered.");
342
343 // If both rooms do not exist, add the first room. If that fails, abort.
344 if (!fromRoomExists && !AddFirstRoom(fromNode))
345 return false;
346
347 // Try to add a room for the edge.
348 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
349
350 // If an edge room is required and was not added, abort.
351 if (!addedEdgeRoom && addEdgeRoom && edge.RequireRoom)
352 return false;
353
354 // Try to add a room for the to node.
355 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
356 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
357 }

384 private bool AddRoom(IRoomSource source,
385 Uid fromRoomId, DoorCode code, EdgeDirection direction)
386 {
387 var fromRoom = Layout.Rooms[fromRoomId];
388 var z = source.Z - fromRoom.Position.Z;
389
390 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
391 {
392 foreach (var config in GetConfigurations(fromRoom.Template, entry.Template))
393 {
394 var position = config.Position + (Vector2DInt)fromRoom.Position;
395
396 // Check if the configuration can be added to the layout. If not, try the next one.
397 if (!config.Matches(z, code, direction))
398 continue;
399 if (Layout.Intersects(entry.Template, position, source.Z))
400 continue;
401
402 // Add the room to the layout.
403 var room = new Room(source, position, entry.Template, RandomSeed);
404 Layout.Rooms.Add(room.Id, room);
405
406 // Try to create a door connection between the rooms.
407 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
408 {
409 // Remove the room added previously.
410 Layout.Rooms.Remove(room.Id);
411 continue;
412 }
413
414 Layout.IncreaseTemplateCount(entry);
415 return true;
416 }
417 }
418
419 return false;
420 }

432 private bool InsertRoom(IRoomSource source,
433 Uid backRoomId, DoorCode backCode, EdgeDirection backDirection,
434 Uid aheadRoomId, DoorCode aheadCode, EdgeDirection aheadDirection)
435 {
436 var backRoom = Layout.Rooms[backRoomId];
437 var aheadRoom = Layout.Rooms[aheadRoomId];
438 var z1 = source.Z - backRoom.Position.Z;
439 var z2 = aheadRoom.Position.Z - source.Z;
440
441 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
442 {
443 foreach (var config1 in GetConfigurations(backRoom.Template, entry.Template))
444 {
445 var position = config1.Position + (Vector2DInt)backRoom.Position;
446
447 // Check if the configuration can be added to the layout. If not, try the next one.
448 if (!config1.Matches(z1, backCode, backDirection))
449 continue;
450 if (Layout.Intersects(entry.Template, position, source.Z))
451 continue;
452
453 foreach (var config2 in GetConfigurations(entry.Template, aheadRoom.Template))
454 {
455 var offset = (Vector2DInt)aheadRoom.Position - position;
456
457 // Check if the configuration can be added to the layout. If not, try the next one.
458 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
459 continue;
460
461 var room = new Room(source, position, entry.Template, RandomSeed);
462 Layout.Rooms.Add(room.Id, room);
463
464 // Try to create a door connection between the rooms.
465 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
466 {
467 // Remove the room added previously.
468 Layout.Rooms.Remove(room.Id);
469 continue;
470 }
471
472 // Try to create a door connection between the rooms.
473 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
474 {
475 // Remove the room and door connection added previously.
476 Layout.Rooms.Remove(room.Id);
477 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
478 continue;
479 }
480
481 Layout.IncreaseTemplateCount(entry);
482 return true;
483 }
484 }
485 }
486
487 return false;

497 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
498 {
499 var fromRoom = Layout.Rooms[fromRoomId];
500 var toRoom = Layout.Rooms[toRoomId];
501 var fromDoor = config.FromDoor;
502 var toDoor = config.ToDoor;
503 Box shaft = null;
504
505 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) > 1)
506 {
507 // Shaft is required.
508 var position = fromDoor.Position + (Vector2DInt)fromRoom.Position;
509 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
510 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
511 var min = new Vector3DInt(position.X, position.Y, zMin);
512 var max = new Vector3DInt(position.X, position.Y, zMax);
513
514 // If shaft intersects the layout, abort adding the door connection.
515 if (Layout.Intersects(min, max))
516 return false;
517
518 shaft = new Box(min, max);
519 }
520
521 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
522 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
523 return true;
524 }
