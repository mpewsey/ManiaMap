168 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups,
169 RandomSeed randomSeed, Action<string> logger = null, CancellationToken cancellationToken = default)
170 {
171 logger?.Invoke("[Layout Generator] Running layout generator...");
172 Initialize(graph, templateGroups, randomSeed);
173
174 var chains = Graph.FindChains(MaxBranchLength);
175 var baseLayout = new Layout(layoutId, graph.Name, randomSeed.Seed);
176 var layouts = new Stack<Layout>(chains.Count);
177 layouts.Push(baseLayout);
178
179 while (layouts.Count > 0)
180 {
181 // Check cancellation token.
182 if (cancellationToken.IsCancellationRequested)
183 {
184 logger?.Invoke("[Layout Generator] Process cancelled.");
185 return null;
186 }
187
188 Layout = layouts.Peek();
189
190 // If all chains have been added, validate the layout.
191 if (ChainIndex >= chains.Count)
192 {
193 // If layout is complete, return the layout.
194 if (Layout.IsComplete(TemplateGroups))
195 {
196 Layout = new Layout(Layout);
197 logger?.Invoke("[Layout Generator] Layout generator complete.");
198 return Layout;
199 }
200
201 // If layout is not complete, start over.
202 ChainIndex = 0;
203 layouts.Clear();
204 layouts.Push(baseLayout);
205 logger?.Invoke("[Layout Generator] Layout constraints not satisfied. Restarting...");
206 continue;
207 }
208
209 // If layout has been used as a base more than the maximum allowable count,
210 // remove the layout and backtrack.
211 if (Layout.Rebases > AllowableRebases)
212 {
213 layouts.Pop();
214 ChainIndex--;
215 logger?.Invoke("[Layout Generator] Rebase count exceeded. Backtracking...");
216 continue;
217 }
218
219 // Create a new layout and try to add the next chain.
220 Layout = new Layout(Layout);
221
222 if (AddChain(chains[ChainIndex]))
223 {
224 layouts.Push(Layout);
225 ChainIndex++;
226 logger?.Invoke($"[Layout Generator] Added chain {ChainIndex} / {chains.Count}...");
227 }
228 }
229
230 logger?.Invoke("[Layout Generator] Layout generator failed.");
231 return null;
232 }

270 private bool AddChain(List<LayoutEdge> chain)
271 {
272 for (int i = 0; i < chain.Count; i++)
273 {
274 var backEdge = chain[i];
275 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
276
277 if (CanInsertRoom(backEdge, aheadEdge))
278 {
279 if (!InsertRooms(backEdge, aheadEdge))
280 return false;
281 i++;
282 }
283 else if (!AddRooms(backEdge))
284 {
285 return false;
286 }
287 }
288
289 return true;
290 }

297 private bool CanInsertRoom(LayoutEdge backEdge, LayoutEdge aheadEdge)
298 {
299 return aheadEdge != null

311 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
312 {
313 var midNode = Graph.GetNode(backEdge.ToNode);
314 var backNode = Graph.GetNode(backEdge.FromNode);
315 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
316
317 // Try to add edge rooms.
318 var addBackEdgeRoom = RandomSeed.ChanceSatisfied(backEdge.RoomChance);
319 var addAheadEdgeRoom = RandomSeed.ChanceSatisfied(aheadEdge.RoomChance);
320 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
321 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
322
323 // If a back edge room is required and was not added, abort.
324 if (!addedBackEdgeRoom && addBackEdgeRoom && backEdge.RequireRoom)
325 return false;
326
327 // If an ahead edge room is required and was not added, abort.
328 if (!addedAheadEdgeRoom && addAheadEdgeRoom && aheadEdge.RequireRoom)
329 return false;
330
331 // Try to insert the node room.
332 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
333 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
334
335 return InsertRoom(midNode,
336 backRoomId, backEdge.DoorCode, backEdge.Direction,
337 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
338 }

345 private bool AddRooms(LayoutEdge edge)
346 {
347 var fromNode = Graph.GetNode(edge.FromNode);
348 var toNode = Graph.GetNode(edge.ToNode);
349 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
350 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
351 var addEdgeRoom = RandomSeed.ChanceSatisfied(edge.RoomChance);
352
353 if (toRoomExists)
354 throw new InvalidChainOrderException("To Node Exists. Chains are not properly ordered.");
355
356 // If both rooms do not exist, add the first room. If that fails, abort.
357 if (!fromRoomExists && !AddFirstRoom(fromNode))
358 return false;
359
360 // Try to add a room for the edge.
361 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
362
363 // If an edge room is required and was not added, abort.
364 if (!addedEdgeRoom && addEdgeRoom && edge.RequireRoom)
365 return false;
366
367 // Try to add a room for the to node.
368 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
369 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
370 }

397 private bool AddRoom(IRoomSource source,
398 Uid fromRoomId, DoorCode code, EdgeDirection direction)
399 {
400 var fromRoom = Layout.Rooms[fromRoomId];
401 var z = source.Z - fromRoom.Position.Z;
402
403 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
404 {
405 foreach (var config in GetConfigurations(fromRoom.Template, entry.Template))
406 {
407 var position = config.Position + fromRoom.Position.To2D();
408
409 // Check if the configuration can be added to the layout. If not, try the next one.
410 if (!config.Matches(z, code, direction))
411 continue;
412 if (Layout.Intersects(entry.Template, position, source.Z))
413 continue;
414
415 // Add the room to the layout.
416 var room = new Room(source, position, entry.Template, RandomSeed.Next());
417 Layout.Rooms.Add(room.Id, room);
418
419 // Try to create a door connection between the rooms.
420 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
421 {
422 // Remove the room added previously.
423 Layout.Rooms.Remove(room.Id);
424 continue;
425 }
426
427 Layout.IncreaseTemplateCount(entry);
428 return true;
429 }
430 }
431
432 return false;
433 }

445 private bool InsertRoom(IRoomSource source,
446 Uid backRoomId, DoorCode backCode, EdgeDirection backDirection,
447 Uid aheadRoomId, DoorCode aheadCode, EdgeDirection aheadDirection)
448 {
449 var backRoom = Layout.Rooms[backRoomId];
450 var aheadRoom = Layout.Rooms[aheadRoomId];
451 var z1 = source.Z - backRoom.Position.Z;
452 var z2 = aheadRoom.Position.Z - source.Z;
453
454 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
455 {
456 foreach (var config1 in GetConfigurations(backRoom.Template, entry.Template))
457 {
458 var position = config1.Position + backRoom.Position.To2D();
459
460 // Check if the configuration can be added to the layout. If not, try the next one.
461 if (!config1.Matches(z1, backCode, backDirection))
462 continue;
463 if (Layout.Intersects(entry.Template, position, source.Z))
464 continue;
465
466 foreach (var config2 in GetConfigurations(entry.Template, aheadRoom.Template))
467 {
468 var offset = aheadRoom.Position.To2D() - position;
469
470 // Check if the configuration can be added to the layout. If not, try the next one.
471 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
472 continue;
473
474 var room = new Room(source, position, entry.Template, RandomSeed.Next());
475 Layout.Rooms.Add(room.Id, room);
476
477 // Try to create a door connection between the rooms.
478 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
479 {
480 // Remove the room added previously.
481 Layout.Rooms.Remove(room.Id);
482 continue;
483 }
484
485 // Try to create a door connection between the rooms.
486 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
487 {
488 // Remove the room and door connection added previously.
489 Layout.Rooms.Remove(room.Id);
490 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
491 continue;
492 }
493
494 Layout.IncreaseTemplateCount(entry);
495 return true;
496 }
497 }
498 }
499
500 return false;

510 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
511 {
512 var fromRoom = Layout.Rooms[fromRoomId];
513 var toRoom = Layout.Rooms[toRoomId];
514 var fromDoor = config.FromDoor;
515 var toDoor = config.ToDoor;
516 Box shaft = null;
517
518 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) > 1)
519 {
520 // Shaft is required.
521 var position = fromDoor.Position + fromRoom.Position.To2D();
522 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
523 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
524 var min = new Vector3DInt(position.X, position.Y, zMin);
525 var max = new Vector3DInt(position.X, position.Y, zMax);
526
527 // If shaft intersects the layout, abort adding the door connection.
528 if (Layout.Intersects(min, max))
529 return false;
530
531 shaft = new Box(min, max);
532 }
533
534 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
535 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
536 return true;
537 }
