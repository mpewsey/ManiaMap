158 public Layout Generate(int layoutId, LayoutGraph graph, TemplateGroups templateGroups, RandomSeed randomSeed)
159 {
160 Initialize(graph, templateGroups, randomSeed);
161
162 var chains = Graph.FindChains(MaxBranchLength);
163 var baseLayout = new Layout(layoutId, graph.Name, randomSeed);
164 var layouts = new Stack<Layout>(chains.Count);
165 layouts.Push(baseLayout);
166
167 while (layouts.Count > 0)
168 {
169 Layout = layouts.Peek();
170
171 // If all chains have been added, validate the layout.
172 if (ChainIndex >= chains.Count)
173 {
174 // If layout is complete, return the layout.
175 if (Layout.IsComplete(TemplateGroups))
176 return new Layout(Layout);
177
178 // If layout is not complete, start over.
179 ChainIndex = 0;
180 layouts.Clear();
181 layouts.Push(baseLayout);
182 continue;
183 }
184
185 // If layout has been used as a base more than the maximum allowable count,
186 // remove the layout and backtrack.
187 if (Layout.Rebases > AllowableRebases)
188 {
189 layouts.Pop();
190 ChainIndex--;
191 continue;
192 }
193
194 // Create a new layout and try to add the next chain.
195 Layout = new Layout(Layout);
196
197 if (AddChain(chains[ChainIndex]))
198 {
199 layouts.Push(Layout);
200 ChainIndex++;
201 }
202 }
203
204 return null;
205 }

243 private bool AddChain(List<LayoutEdge> chain)
244 {
245 for (int i = 0; i < chain.Count; i++)
246 {
247 var backEdge = chain[i];
248 var aheadEdge = i + 1 < chain.Count ? chain[i + 1] : null;
249
250 if (CanInsertRoom(backEdge, aheadEdge))
251 {
252 if (!InsertRooms(backEdge, aheadEdge))
253 return false;
254 i++;
255 }
256 else if (!AddRooms(backEdge))
257 {
258 return false;
259 }
260 }
261
262 return true;

284 private bool InsertRooms(LayoutEdge backEdge, LayoutEdge aheadEdge)
285 {
286 var midNode = Graph.GetNode(backEdge.ToNode);
287 var backNode = Graph.GetNode(backEdge.FromNode);
288 var aheadNode = Graph.GetNode(aheadEdge.ToNode);
289
290 // Try to add edge rooms.
291 var addBackEdgeRoom = backEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
292 var addAheadEdgeRoom = aheadEdge.RoomChanceSatisfied(RandomSeed.NextDouble());
293 var addedBackEdgeRoom = addBackEdgeRoom && AddRoom(backEdge, backNode.RoomId, backEdge.DoorCode, backEdge.Direction);
294 var addedAheadEdgeRoom = addAheadEdgeRoom && AddRoom(aheadEdge, aheadNode.RoomId, aheadEdge.DoorCode, LayoutEdge.ReverseEdgeDirection(aheadEdge.Direction));
295
296 // If a back edge room is required and was not added, abort.
297 if (!addedBackEdgeRoom && addBackEdgeRoom && backEdge.RequireRoom)
298 return false;
299
300 // If an ahead edge room is required and was not added, abort.
301 if (!addedAheadEdgeRoom && addAheadEdgeRoom && aheadEdge.RequireRoom)
302 return false;
303
304 // Try to insert the node room.
305 var backRoomId = addedBackEdgeRoom ? backEdge.RoomId : backNode.RoomId;
306 var aheadRoomId = addedAheadEdgeRoom ? aheadEdge.RoomId : aheadNode.RoomId;
307
308 return InsertRoom(midNode,
309 backRoomId, backEdge.DoorCode, backEdge.Direction,
310 aheadRoomId, aheadEdge.DoorCode, aheadEdge.Direction);
311 }

318 private bool AddRooms(LayoutEdge edge)
319 {
320 var fromNode = Graph.GetNode(edge.FromNode);
321 var toNode = Graph.GetNode(edge.ToNode);
322 var fromRoomExists = Layout.Rooms.ContainsKey(fromNode.RoomId);
323 var toRoomExists = Layout.Rooms.ContainsKey(toNode.RoomId);
324 var addEdgeRoom = edge.RoomChanceSatisfied(RandomSeed.NextDouble());
325
326 if (toRoomExists)
327 throw new InvalidChainOrderException("To Node Exists. Chains are not properly ordered.");
328
329 // If both rooms do not exist, add the first room. If that fails, abort.
330 if (!fromRoomExists && !AddFirstRoom(fromNode))
331 return false;
332
333 // Try to add a room for the edge.
334 var addedEdgeRoom = addEdgeRoom && AddRoom(edge, fromNode.RoomId, edge.DoorCode, edge.Direction);
335
336 // If an edge room is required and was not added, abort.
337 if (!addedEdgeRoom && addEdgeRoom && edge.RequireRoom)
338 return false;
339
340 // Try to add a room for the to node.
341 var fromRoomId = addedEdgeRoom ? edge.RoomId : fromNode.RoomId;
342 return AddRoom(toNode, fromRoomId, edge.DoorCode, edge.Direction);
343 }

370 private bool AddRoom(IRoomSource source,
371 Uid fromRoomId, int code, EdgeDirection direction)
372 {
373 var fromRoom = Layout.Rooms[fromRoomId];
374 var z = source.Z - fromRoom.Position.Z;
375
376 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
377 {
378 foreach (var config in GetConfigurations(fromRoom.Template, entry.Template))
379 {
380 var position = config.Position + fromRoom.Position;
381
382 // Check if the configuration can be added to the layout. If not, try the next one.
383 if (!config.Matches(z, code, direction))
384 continue;
385 if (Layout.Intersects(entry.Template, position, source.Z))
386 continue;
387
388 // Add the room to the layout.
389 var room = new Room(source, position, entry.Template, RandomSeed);
390 Layout.Rooms.Add(room.Id, room);
391
392 // Try to create a door connection between the rooms.
393 if (!AddDoorConnection(fromRoomId, source.RoomId, config))
394 {
395 // Remove the room added previously.
396 Layout.Rooms.Remove(room.Id);
397 continue;
398 }
399
400 Layout.IncreaseTemplateCount(entry);
401 return true;
402 }
403 }
404
405 return false;

418 private bool InsertRoom(IRoomSource source,
419 Uid backRoomId, int backCode, EdgeDirection backDirection,
420 Uid aheadRoomId, int aheadCode, EdgeDirection aheadDirection)
421 {
422 var backRoom = Layout.Rooms[backRoomId];
423 var aheadRoom = Layout.Rooms[aheadRoomId];
424 var z1 = source.Z - backRoom.Position.Z;
425 var z2 = aheadRoom.Position.Z - source.Z;
426
427 foreach (var entry in GetTemplateGroupEntries(source.TemplateGroup))
428 {
429 foreach (var config1 in GetConfigurations(backRoom.Template, entry.Template))
430 {
431 var position = config1.Position + backRoom.Position;
432
433 // Check if the configuration can be added to the layout. If not, try the next one.
434 if (!config1.Matches(z1, backCode, backDirection))
435 continue;
436 if (Layout.Intersects(entry.Template, position, source.Z))
437 continue;
438
439 foreach (var config2 in GetConfigurations(entry.Template, aheadRoom.Template))
440 {
441 var offset = aheadRoom.Position - position;
442
443 // Check if the configuration can be added to the layout. If not, try the next one.
444 if (!config2.Matches(offset, z2, aheadCode, aheadDirection))
445 continue;
446
447 var room = new Room(source, position, entry.Template, RandomSeed);
448 Layout.Rooms.Add(room.Id, room);
449
450 // Try to create a door connection between the rooms.
451 if (!AddDoorConnection(backRoomId, source.RoomId, config1))
452 {
453 // Remove the room added previously.
454 Layout.Rooms.Remove(room.Id);
455 continue;
456 }
457
458 // Try to create a door connection between the rooms.
459 if (!AddDoorConnection(source.RoomId, aheadRoomId, config2))
460 {
461 // Remove the room and door connection added previously.
462 Layout.Rooms.Remove(room.Id);
463 Layout.RemoveDoorConnection(backRoomId, source.RoomId);
464 continue;
465 }
466
467 Layout.IncreaseTemplateCount(entry);
468 return true;
469 }
470 }
471 }
472
473 return false;

483 private bool AddDoorConnection(Uid fromRoomId, Uid toRoomId, Configuration config)
484 {
485 var fromRoom = Layout.Rooms[fromRoomId];
486 var toRoom = Layout.Rooms[toRoomId];
487 var fromDoor = config.FromDoor;
488 var toDoor = config.ToDoor;
489 Box shaft = null;
490
491 if (Math.Abs(fromRoom.Position.Z - toRoom.Position.Z) > 1)
492 {
493 // Shaft is required.
494 var position = fromDoor.Position + fromRoom.Position;
495 var zMin = Math.Min(fromRoom.Position.Z, toRoom.Position.Z) + 1;
496 var zMax = Math.Max(fromRoom.Position.Z, toRoom.Position.Z) - 1;
497 var min = new Vector3DInt(position.X, position.Y, zMin);
498 var max = new Vector3DInt(position.X, position.Y, zMax);
499
500 // If shaft intersects the layout, abort adding the door connection.
501 if (Layout.Intersects(min, max))
502 return false;
503
504 shaft = new Box(min, max);
505 }
506
507 var connection = new DoorConnection(fromRoom, toRoom, fromDoor, toDoor, shaft);
508 Layout.DoorConnections.Add(new RoomPair(fromRoomId, toRoomId), connection);
509 return true;
510 }
