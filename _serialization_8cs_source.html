148 public static string DecryptTextFile(string path, byte[] key)
149 {
150 using (var stream = File.OpenRead(path))
151 using (var algorithm = Aes.Create())
152 {
153 var iv = new byte[algorithm.IV.Length];
154 stream.Read(iv, 0, iv.Length);
155
156 using (var encryptor = algorithm.CreateDecryptor(key, iv))
157 using (var crypto = new CryptoStream(stream, encryptor, CryptoStreamMode.Read))
