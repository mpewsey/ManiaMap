409 private List<DoorPair> ShaftAlignedDoors(RoomTemplate other, Vector2DInt offset)
410 {
411 var result = new List<DoorPair>();
412 var jStart = Math.Max(0, offset.Y - 1);
413 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);
414
415 if (jStart >= jStop)
416 return result;
417
418 var iStart = Math.Max(0, offset.X - 1);
419 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
420
421 for (int i = iStart; i < iStop; i++)
422 {
423 for (int j = jStart; j < jStop; j++)
424 {
425 var cell = Cells[i, j];
426
427 if (cell == null)
428 continue;
429
430 var index1 = new Vector2DInt(i, j);
431 var index2 = index1 - offset;
432 var vert = other.Cells.GetOrDefault(index2.X, index2.Y);
433
434 if (cell.TopDoorAligns(vert))
435 {
436 var door1 = new DoorPosition(index1, DoorDirection.Top, cell.TopDoor);
437 var door2 = new DoorPosition(index2, DoorDirection.Bottom, vert.BottomDoor);
438 result.Add(new DoorPair(door1, door2));
439 }
440
441 if (cell.BottomDoorAligns(vert))
442 {
443 var door1 = new DoorPosition(index1, DoorDirection.Bottom, cell.BottomDoor);
444 var door2 = new DoorPosition(index2, DoorDirection.Top, vert.TopDoor);
445 result.Add(new DoorPair(door1, door2));
446 }
447 }
448 }
449
450 return result;

459 private List<DoorPair> PlaneAlignedDoors(RoomTemplate other, Vector2DInt offset)
460 {
461 var result = new List<DoorPair>();
462 var jStart = Math.Max(0, offset.Y - 1);
463 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);
464
465 if (jStart >= jStop)
466 return result;
467
468 var iStart = Math.Max(0, offset.X - 1);
469 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
470
471 for (int i = iStart; i < iStop; i++)
472 {
473 for (int j = jStart; j < jStop; j++)
474 {
475 var cell = Cells[i, j];
476
477 if (cell == null)
478 continue;
479
480 var index = new Vector2DInt(i, j);
481 var x = i - offset.X;
482 var y = j - offset.Y;
483
484 var west = other.Cells.GetOrDefault(x, y - 1);
485 var north = other.Cells.GetOrDefault(x - 1, y);
486 var east = other.Cells.GetOrDefault(x, y + 1);
487 var south = other.Cells.GetOrDefault(x + 1, y);
488
489 if (cell.WestDoorAligns(west))
490 {

492 var door2 = new DoorPosition(new Vector2DInt(x, y - 1), DoorDirection.East, west.EastDoor);
493 result.Add(new DoorPair(door1, door2));
494 }
495
496 if (cell.NorthDoorAligns(north))
