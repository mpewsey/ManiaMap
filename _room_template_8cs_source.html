421 private List<DoorPair> ShaftAlignedDoors(RoomTemplate other, Vector2DInt offset)
422 {
423 var result = new List<DoorPair>();
424 var jStart = Math.Max(0, offset.Y - 1);
425 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);
426
427 if (jStart >= jStop)
428 return result;
429
430 var iStart = Math.Max(0, offset.X - 1);
431 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
432
433 for (int i = iStart; i < iStop; i++)
434 {
435 for (int j = jStart; j < jStop; j++)
436 {
437 var cell = Cells[i, j];
438
439 if (cell == null)
440 continue;
441
442 var index1 = new Vector2DInt(i, j);
443 var index2 = index1 - offset;
444 var vert = other.Cells.GetOrDefault(index2.X, index2.Y);
445
446 if (cell.TopDoorAligns(vert))
447 {
448 var door1 = new DoorPosition(index1, DoorDirection.Top, cell.TopDoor);
449 var door2 = new DoorPosition(index2, DoorDirection.Bottom, vert.BottomDoor);
450 result.Add(new DoorPair(door1, door2));
451 }
452
453 if (cell.BottomDoorAligns(vert))
454 {
455 var door1 = new DoorPosition(index1, DoorDirection.Bottom, cell.BottomDoor);
456 var door2 = new DoorPosition(index2, DoorDirection.Top, vert.TopDoor);
457 result.Add(new DoorPair(door1, door2));

471 private List<DoorPair> PlaneAlignedDoors(RoomTemplate other, Vector2DInt offset)

475 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);

478 return result;
479
480 var iStart = Math.Max(0, offset.X - 1);
481 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
482
483 for (int i = iStart; i < iStop; i++)
484 {
485 for (int j = jStart; j < jStop; j++)
486 {
487 var cell = Cells[i, j];
488
489 if (cell == null)
490 continue;
491
492 var index = new Vector2DInt(i, j);
493 var x = i - offset.X;
494 var y = j - offset.Y;
495
496 var west = other.Cells.GetOrDefault(x, y - 1);
497 var north = other.Cells.GetOrDefault(x - 1, y);
498 var east = other.Cells.GetOrDefault(x, y + 1);
499 var south = other.Cells.GetOrDefault(x + 1, y);
500
501 if (cell.WestDoorAligns(west))

504 var door2 = new DoorPosition(new Vector2DInt(x, y - 1), DoorDirection.East, west.EastDoor);

506 }

508 if (cell.NorthDoorAligns(north))
