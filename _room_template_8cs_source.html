550 public bool Intersects(Vector2DInt min, Vector2DInt max)
551 {
552 var jStart = Math.Max(min.X, 0);
553 var jStop = Math.Min(max.X + 1, Cells.Rows);
554
555 if (jStart < jStop)
556 {
557 var iStart = Math.Max(min.Y, 0);
558 var iStop = Math.Min(max.Y + 1, Cells.Columns);
559
560 for (int i = iStart; i < iStop; i++)
561 {
562 for (int j = jStart; j < jStop; j++)
563 {
564 if (Cells[i, j] != null)
565 return true;
566 }
567 }
568 }
569
570 return false;
571 }

607 private List<DoorPair> ShaftAlignedDoors(RoomTemplate other, Vector2DInt offset)
608 {
609 var result = new List<DoorPair>();
610 var jStart = Math.Max(0, offset.Y - 1);
611 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);
612
613 if (jStart >= jStop)
614 return result;
615
616 var iStart = Math.Max(0, offset.X - 1);
617 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
618
619 for (int i = iStart; i < iStop; i++)
620 {
621 for (int j = jStart; j < jStop; j++)
622 {
623 var cell = Cells[i, j];
624
625 if (cell == null)
626 continue;
627
628 var index1 = new Vector2DInt(i, j);
629 var index2 = index1 - offset;
630 var vert = other.Cells.GetOrDefault(index2.X, index2.Y);
631
632 if (cell.TopDoorAligns(vert))
633 {
634 var door1 = new DoorPosition(index1, DoorDirection.Top, cell.TopDoor);
635 var door2 = new DoorPosition(index2, DoorDirection.Bottom, vert.BottomDoor);
636 result.Add(new DoorPair(door1, door2));
637 }
638
639 if (cell.BottomDoorAligns(vert))
640 {
641 var door1 = new DoorPosition(index1, DoorDirection.Bottom, cell.BottomDoor);
642 var door2 = new DoorPosition(index2, DoorDirection.Top, vert.TopDoor);
643 result.Add(new DoorPair(door1, door2));
644 }
645 }
646 }
647
648 return result;
649 }

657 private List<DoorPair> PlaneAlignedDoors(RoomTemplate other, Vector2DInt offset)
658 {
659 var result = new List<DoorPair>();
660 var jStart = Math.Max(0, offset.Y - 1);
661 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);
662
663 if (jStart >= jStop)
664 return result;
665
666 var iStart = Math.Max(0, offset.X - 1);
667 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
668
669 for (int i = iStart; i < iStop; i++)
670 {
671 for (int j = jStart; j < jStop; j++)
672 {
673 var cell = Cells[i, j];
674
675 if (cell == null)
676 continue;
677
678 var index = new Vector2DInt(i, j);
679 var x = i - offset.X;
680 var y = j - offset.Y;
681
682 var west = other.Cells.GetOrDefault(x, y - 1);
683 var north = other.Cells.GetOrDefault(x - 1, y);
684 var east = other.Cells.GetOrDefault(x, y + 1);
685 var south = other.Cells.GetOrDefault(x + 1, y);
686
687 if (cell.WestDoorAligns(west))
688 {
689 var door1 = new DoorPosition(index, DoorDirection.West, cell.WestDoor);
690 var door2 = new DoorPosition(new Vector2DInt(x, y - 1), DoorDirection.East, west.EastDoor);
691 result.Add(new DoorPair(door1, door2));
692 }
693
694 if (cell.NorthDoorAligns(north))
695 {
696 var door1 = new DoorPosition(index, DoorDirection.North, cell.NorthDoor);
697 var door2 = new DoorPosition(new Vector2DInt(x - 1, y), DoorDirection.South, north.SouthDoor);
698 result.Add(new DoorPair(door1, door2));
699 }
700
701 if (cell.EastDoorAligns(east))
702 {
703 var door1 = new DoorPosition(index, DoorDirection.East, cell.EastDoor);
704 var door2 = new DoorPosition(new Vector2DInt(x, y + 1), DoorDirection.West, east.WestDoor);
705 result.Add(new DoorPair(door1, door2));
706 }
707
708 if (cell.SouthDoorAligns(south))
709 {
710 var door1 = new DoorPosition(index, DoorDirection.South, cell.SouthDoor);

712 result.Add(new DoorPair(door1, door2));

714 }
