416 private List<DoorPair> ShaftAlignedDoors(RoomTemplate other, Vector2DInt offset)
417 {
418 var result = new List<DoorPair>();
419 var jStart = Math.Max(0, offset.Y - 1);
420 var jStop = Math.Min(Cells.Columns, other.Cells.Columns + offset.Y + 1);
421
422 if (jStart >= jStop)
423 return result;
424
425 var iStart = Math.Max(0, offset.X - 1);
426 var iStop = Math.Min(Cells.Rows, other.Cells.Rows + offset.X + 1);
427
428 for (int i = iStart; i < iStop; i++)
429 {
430 for (int j = jStart; j < jStop; j++)
431 {
432 var cell = Cells[i, j];
433
434 if (cell == null)
435 continue;
436
437 var index1 = new Vector2DInt(i, j);
438 var index2 = index1 - offset;
439 var vert = other.Cells.GetOrDefault(index2.X, index2.Y);
440
441 if (cell.TopDoorAligns(vert))
442 {
443 var door1 = new DoorPosition(index1, DoorDirection.Top, cell.TopDoor);
444 var door2 = new DoorPosition(index2, DoorDirection.Bottom, vert.BottomDoor);
445 result.Add(new DoorPair(door1, door2));
446 }
447
448 if (cell.BottomDoorAligns(vert))
449 {
450 var door1 = new DoorPosition(index1, DoorDirection.Bottom, cell.BottomDoor);
451 var door2 = new DoorPosition(index2, DoorDirection.Top, vert.TopDoor);
452 result.Add(new DoorPair(door1, door2));
453 }

466 private List<DoorPair> PlaneAlignedDoors(RoomTemplate other, Vector2DInt offset)

479 {
480 for (int j = jStart; j < jStop; j++)

482 var cell = Cells[i, j];
483
484 if (cell == null)
485 continue;
486

488 var x = i - offset.X;
489 var y = j - offset.Y;

492 var north = other.Cells.GetOrDefault(x - 1, y);

499 var door2 = new DoorPosition(new Vector2DInt(x, y - 1), DoorDirection.East, west.EastDoor);
