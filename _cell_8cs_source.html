217 case 't':

436 public bool ValuesAreEqual(Cell other)
437 {
438 if (Doors.Count != other.Doors.Count
439 || CollectableSpots.Count != other.CollectableSpots.Count
440 || Features.Count != other.Features.Count)
441 return false;
442
443 foreach (var pair in other.Doors)
444 {
445 if (!Doors.TryGetValue(pair.Key, out Door value)
446 || !Door.ValuesAreEqual(pair.Value, value))
447 return false;
448 }
449
450 foreach (var pair in other.CollectableSpots)
451 {
452 if (!CollectableSpots.TryGetValue(pair.Key, out string value)
453 || pair.Value != value)
454 return false;
455 }
456
457 for (int i = 0; i < Features.Count; i++)
458 {
459 if (Features[i] != other.Features[i])
460 return false;
461 }
462
463 return true;
464 }
