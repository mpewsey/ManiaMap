405 public bool ValuesAreEqual(Cell other)
406 {
407 if (Doors.Count != other.Doors.Count
408 || CollectableSpots.Count != other.CollectableSpots.Count
409 || Features.Count != other.Features.Count)
410 return false;
411
412 foreach (var pair in other.Doors)
413 {
414 if (!Doors.TryGetValue(pair.Key, out Door value)
415 || !Door.ValuesAreEqual(pair.Value, value))
416 return false;
417 }
418
419 foreach (var pair in other.CollectableSpots)
420 {
421 if (!CollectableSpots.TryGetValue(pair.Key, out string value)
422 || pair.Value != value)
423 return false;
424 }
425
426 for (int i = 0; i < Features.Count; i++)
427 {
428 if (Features[i] != other.Features[i])
429 return false;
430 }
431
432 return true;
433 }
