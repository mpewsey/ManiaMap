420 public bool ValuesAreEqual(Cell other)
421 {
422 if (Doors.Count != other.Doors.Count || CollectableSpots.Count != other.CollectableSpots.Count)
423 return false;
424
425 foreach (var pair in other.Doors)
426 {
427 if (!Doors.TryGetValue(pair.Key, out Door value) || !Door.ValuesAreEqual(pair.Value, value))

429 return false;
