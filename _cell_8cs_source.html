133 public Cell SetDoors(string directions, Door door)
134 {
135 foreach (var direction in directions)
136 {
137 switch (direction)
138 {
139 case 'N':
140 case 'n':
141 NorthDoor = door?.Copy();
142 break;
143 case 'S':
144 case 's':
145 SouthDoor = door?.Copy();
146 break;
147 case 'W':
148 case 'w':
149 WestDoor = door?.Copy();
150 break;
151 case 'E':
152 case 'e':
153 EastDoor = door?.Copy();
154 break;
155 case 'B':
156 case 'b':
157 BottomDoor = door?.Copy();
158 break;
159 case 'T':
160 case 't':
161 TopDoor = door?.Copy();
162 break;
163 default:
164 throw new UnhandledCaseException($"Unhandled character: {direction}.");
165 }
166 }

168 return this;

178 public Cell AddCollectableGroup(int locationId, string group)
179 {
180 if (string.IsNullOrWhiteSpace(group))
181 throw new InvalidNameException($"Group name is null or white space.");
182
183 var index = CollectableGroups.FindIndex(x => x.Id == locationId);

185 if (index >= 0)
