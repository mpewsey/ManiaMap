371 public bool ValuesAreEqual(Cell other)
372 {
373 if (Doors.Count != other.Doors.Count)
374 return false;
375
376 if (CollectableSpots.Count != other.CollectableSpots.Count)
377 return false;
378
379 foreach (var pair in Doors)
380 {
381 if (!other.Doors.TryGetValue(pair.Key, out Door value) || !Door.ValuesAreEqual(pair.Value, value))
382 {
383 return false;
384 }
385 }
386
387 foreach (var pair in CollectableSpots)
388 {
389 if (!other.CollectableSpots.TryGetValue(pair.Key, out string value) || pair.Value != value)
390 {
391 return false;
392 }
393 }
394
395 return true;
396 }
