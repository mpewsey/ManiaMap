93 public Cell SetDoors(string directions, Door door)
94 {
95 foreach (var direction in directions)
96 {
97 switch (direction)
98 {
99 case 'N':
100 case 'n':

103 case 'S':
104 case 's':

117 BottomDoor = door?.Copy();
118 break;
119 case 'T':
120 case 't':
121 TopDoor = door?.Copy();
122 break;
123 default:
124 throw new Exception($"Unhandled character: {direction}.");
