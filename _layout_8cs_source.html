272 public RectangleInt GetBounds()
273 {
274 if (Rooms.Count == 0)
275 return new RectangleInt();
276
277 var minX = int.MaxValue;
278 var minY = int.MaxValue;
279 var maxX = int.MinValue;
280 var maxY = int.MinValue;
281
282 foreach (var room in Rooms.Values)
283 {
284 minX = Math.Min(minX, room.Position.X);
285 minY = Math.Min(minY, room.Position.Y);
286 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
287 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
288 }
289
290 return new RectangleInt(minY, minX, maxY - minY, maxX - minX);
291 }
