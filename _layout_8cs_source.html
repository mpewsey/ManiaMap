221 public Rectangle Bounds()
222 {
223 if (Rooms.Count == 0)
224 return new Rectangle();
225
226 var minX = int.MaxValue;
227 var minY = int.MaxValue;
228 var maxX = int.MinValue;
229 var maxY = int.MinValue;
230
231 foreach (var room in Rooms.Values)
232 {
233 minX = Math.Min(minX, room.Position.X);
234 minY = Math.Min(minY, room.Position.Y);
235 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
236 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
237 }
238
239 return new Rectangle(minY, minX, maxY - minY, maxX - minX);
240 }
