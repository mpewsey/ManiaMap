392 public RectangleInt GetBounds()
393 {
394 if (Rooms.Count == 0)
395 return new RectangleInt();
396
397 var minX = int.MaxValue;
398 var minY = int.MaxValue;
399 var maxX = int.MinValue;
400 var maxY = int.MinValue;
401
402 foreach (var room in Rooms.Values)
403 {
404 minX = Math.Min(minX, room.Position.X);
405 minY = Math.Min(minY, room.Position.Y);
406 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
407 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
408 }
409
410 return new RectangleInt(minY, minX, maxY - minY, maxX - minX);
411 }
