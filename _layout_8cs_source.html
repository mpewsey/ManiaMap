389 public RectangleInt GetBounds()
390 {
391 if (Rooms.Count == 0)
392 return new RectangleInt();
393
394 var minX = int.MaxValue;
395 var minY = int.MaxValue;
396 var maxX = int.MinValue;
397 var maxY = int.MinValue;
398
399 foreach (var room in Rooms.Values)
400 {
401 minX = Math.Min(minX, room.Position.X);
402 minY = Math.Min(minY, room.Position.Y);
403 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
404 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
405 }
406
407 return new RectangleInt(minY, minX, maxY - minY, maxX - minX);
408 }
