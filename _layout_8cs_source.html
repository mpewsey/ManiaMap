106 public bool Intersects(Vector3DInt min, Vector3DInt max)
107 {
108 foreach (var room in Rooms.Values)
109 {
110 if (room.Position.Z >= min.Z && room.Position.Z <= max.Z)
111 {
112 if (room.Template.Intersects(min - room.Position, max - room.Position))

254 public Rectangle Bounds()
255 {
256 if (Rooms.Count == 0)
257 return new Rectangle();
258
259 var minX = int.MaxValue;
260 var minY = int.MaxValue;
261 var maxX = int.MinValue;
262 var maxY = int.MinValue;
263
264 foreach (var room in Rooms.Values)
265 {
266 minX = Math.Min(minX, room.Position.X);
267 minY = Math.Min(minY, room.Position.Y);
268 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
269 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
270 }
271
272 return new Rectangle(minY, minX, maxY - minY, maxX - minX);
273 }
