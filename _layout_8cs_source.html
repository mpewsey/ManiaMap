401 public RectangleInt GetBounds()
402 {
403 if (Rooms.Count == 0)
404 return new RectangleInt();
405
406 var minX = int.MaxValue;
407 var minY = int.MaxValue;
408 var maxX = int.MinValue;
409 var maxY = int.MinValue;
410
411 foreach (var room in Rooms.Values)
412 {
413 minX = Math.Min(minX, room.Position.X);
414 minY = Math.Min(minY, room.Position.Y);
415 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
416 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
417 }
418
419 return new RectangleInt(minY, minX, maxY - minY, maxX - minX);
420 }
