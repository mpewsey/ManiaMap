395 public RectangleInt GetBounds()
396 {
397 if (Rooms.Count == 0)
398 return new RectangleInt();
399
400 var minX = int.MaxValue;
401 var minY = int.MaxValue;
402 var maxX = int.MinValue;
403 var maxY = int.MinValue;
404
405 foreach (var room in Rooms.Values)
406 {
407 minX = Math.Min(minX, room.Position.X);
408 minY = Math.Min(minY, room.Position.Y);
409 maxX = Math.Max(maxX, room.Position.X + room.Template.Cells.Rows);
410 maxY = Math.Max(maxY, room.Position.Y + room.Template.Cells.Columns);
411 }
412
413 return new RectangleInt(minY, minX, maxY - minY, maxX - minX);
414 }
