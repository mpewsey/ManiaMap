93 private static void OrderEdgeNodes(Data data)
94 {
95 foreach (var chain in data.Chains)
96 {
97 for (int i = 1; i < chain.Count; i++)
98 {
99 var back = chain[i - 1];
100 var forward = chain[i];
101
102 if (back.FromNode == forward.ToNode)
103 {
104 back.Reverse();
105 forward.Reverse();
106 }
107 else if (back.FromNode == forward.FromNode)
108 {
109 back.Reverse();
110 }
111 else if (back.ToNode == forward.ToNode)
112 {
113 forward.Reverse();
114 }
115 }

160 private static void SplitBrokenChains(Data data)
161 {
162 for (int i = 0; i < data.Chains.Count; i++)
163 {
164 var chain = data.Chains[i];
165
166 for (int j = 1; j < chain.Count; j++)
167 {
168 var x = chain[j - 1];
169 var y = chain[j];
170
171 if (!x.SharesNode(y))
172 {
173 data.Chains[i] = chain.GetRange(0, j);
174 data.Chains.Insert(i + 1, chain.GetRange(j, chain.Count - j));
175 break;
176 }

207 private static List<List<LayoutEdge>> FormSequentialChains(Data data)
208 {
209 var result = new List<List<LayoutEdge>>(data.Chains.Count);
210
211 foreach (var chain in data.Chains)
212 {
213 data.Pool.AddLast(chain);
214 }
215
216 if (data.Pool.First != null)
217 {
218 var chain = data.Pool.First.Value;
219 MarkNodes(data, chain);
220 result.Add(chain);
221 data.Pool.RemoveFirst();
222 }
223
224 for (int i = 1; i < data.Chains.Count; i++)
225 {
226 var chain = FindNextChain(data);
227 MarkNodes(data, chain);
228 result.Add(chain);
229 }
230
231 return result;

238 private static List<LayoutEdge> FindNextChain(Data data)
239 {
240 for (var node = data.Pool.First; node != null; node = node.Next)
241 {
242 var chain = node.Value;
243
244 // If the chain is a cycle, shift the elements of the chain to make a possible sequence.
245 if (ChainIsCycle(chain))
246 {
247 var index = chain.FindIndex(x => data.Marked.Contains(x.FromNode));
248
249 if (index >= 0)
250 {
251 data.Pool.Remove(node);
252 var subChain = chain.GetRange(0, index);
253 chain.RemoveRange(0, index);
254 chain.AddRange(subChain);
255 return chain;
256 }
257
258 continue;
259 }
260
261 // Check if first edge forms sequence.
262 if (data.Marked.Contains(chain[0].FromNode))
263 {
264 data.Pool.Remove(node);
265 return chain;
266 }
267
268 // Check if last edge forms sequence.
269 if (data.Marked.Contains(chain[chain.Count - 1].ToNode))
270 {
271 data.Pool.Remove(node);
272 chain.Reverse();
273 chain.ForEach(x => x.Reverse());
274 return chain;
275 }
276 }
277
278 return null;
279 }
