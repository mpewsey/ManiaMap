82 private void OrderEdgeNodes()
83 {
84 foreach (var chain in Chains)
85 {
86 for (int i = 1; i < chain.Count; i++)
87 {
88 var back = chain[i - 1];
89 var forward = chain[i];
90
91 if (back.FromNode == forward.ToNode)
92 {
93 back.Reverse();
94 forward.Reverse();
95 }
96 else if (back.FromNode == forward.FromNode)
97 {
98 back.Reverse();
99 }
100 else if (back.ToNode == forward.ToNode)
101 {
102 forward.Reverse();
103 }
104 }

147 {

151

191 private List<List<LayoutEdge>> FormSequentialChains()
192 {
193 var result = new List<List<LayoutEdge>>(Chains.Count);
194
195 if (Chains.Count > 0)
196 {
197 var chain = Chains[0];
198 Chains.RemoveAt(0);
199 MarkNodes(chain);
200 result.Add(chain);
201 }
202
203 while (Chains.Count > 0)
204 {
205 var chain = FindNextChain();
206 MarkNodes(chain);
207 result.Add(chain);
208 }
209
210 return result;

216 private List<LayoutEdge> FindNextChain()
217 {
218 for (int i = 0; i < Chains.Count; i++)
219 {
220 var chain = Chains[i];
221
222 // If the chain is a cycle, shift the elements of the chain to make a possible sequence.
223 if (ChainIsCycle(chain))
224 {
225 var index = chain.FindIndex(x => Marked.Contains(x.FromNode));
226
227 if (index >= 0)
228 {
229 Chains.RemoveAt(i);
230 var subChain = chain.GetRange(0, index);
231 chain.RemoveRange(0, index);
232 chain.AddRange(subChain);
233 return chain;
234 }
235
236 continue;
237 }
238
239 // Check if first edge forms sequence.
240 if (Marked.Contains(chain[0].FromNode))
241 {
242 Chains.RemoveAt(i);
243 return chain;
244 }
245
246 // Check if last edge forms sequence.
247 if (Marked.Contains(chain[chain.Count - 1].ToNode))
248 {
249 Chains.RemoveAt(i);
250 chain.Reverse();
251 chain.ForEach(x => x.Reverse());
252 return chain;
253 }
254 }
255
256 return null;
257 }
