45 var paths = new List<List<RoomTemplate>>
46 {
47 square1x1.UniqueVariations(),

49 rect1x3.UniqueVariations(),
50 rect1x4.UniqueVariations(),
51 angle3x4.UniqueVariations(),
52 };
53
54 paths.ForEach(x => templateGroups.Add("Paths", x));
55
56 // Set Default template group to all cells
57 var seed = new RandomSeed(12345);
58
59 foreach (var template in templateGroups.GetAllTemplates())

61 for (int i = 0; i < template.Cells.Rows; i++)
62 {
63 for (int j = 0; j < template.Cells.Columns; j++)
64 {
65 template.AddCollectableSpot(seed.Next(), new Vector2DInt(i, j), "Default");
66 }
67 }
68 }

71 templateGroups.Add("SavePoint", savePoint);
72
73 return templateGroups;
