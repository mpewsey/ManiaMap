62 private void CycleSearch(int node, int parent)
63 {
64 Colors.TryGetValue(node, out var color);
65
66 // Tree traversal from this node is already complete.
67 if (color == 2)
68 return;
69
70 // Tree traversal from this node is in progress.
71 // Therefore, the tree has looped back around, forming a cycle.
72 if (color == 1)
73 {
74 var current = parent;
75 var cycle = new List<int> { current };
76 Cycles.Add(cycle);
77
78 // Accumulate parents into cycle until origin node is encountered.
79 while (current != node)
80 {
81 current = Parents[current];
82 cycle.Add(current);
83 }
84
85 return;
86 }
87
88 // Change color to indicate search is in progress for node.
89 Colors[node] = 1;
90 Parents[node] = parent;
91
92 foreach (var neighbor in Graph.GetNeighbors(node))
93 {
94 if (neighbor != Parents[node])
95 CycleSearch(neighbor, node);
96 }
97
98 // Change color to indicate search from node is complete.
99 Colors[node] = 2;
100 }

106 {

110 foreach (var cycle in Cycles)
