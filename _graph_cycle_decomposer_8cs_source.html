74 private void CycleSearch(T node, T parent)
75 {
76 var comparer = EqualityComparer<T>.Default;
77 Colors.TryGetValue(node, out var color);
78
79 // Tree traversal from this node is already complete.
80 if (color == 2)
81 return;
82
83 // Tree traversal from this node is in progress.
84 // Therefore, the tree has looped back around, forming a cycle.
85 if (color == 1)
86 {
87 var current = parent;
88 var cycle = new List<T> { current };
89 Cycles.Add(cycle);
90
91 // Accumulate parents into cycle until origin node is encountered.
92 while (!comparer.Equals(current, node))
93 {
94 current = Parents[current];
95 cycle.Add(current);
96 }
97
98 return;
99 }
100
101 // Change color to indicate search is in progress for node.
102 Colors[node] = 1;
103 Parents[node] = parent;
104
105 foreach (var neighbor in Neighbors[node])
106 {
107 if (!comparer.Equals(neighbor, Parents[node]))
108 CycleSearch(neighbor, node);
109 }
110
111 // Change color to indicate search from node is complete.
112 Colors[node] = 2;
113 }

118 private List<List<T>> GetUniqueCycles()
119 {
120 var sets = new List<HashSet<T>>();
121 var result = new List<List<T>>();
122
123 foreach (var cycle in Cycles)
124 {
125 if (!sets.Any(x => x.SetEquals(cycle)))
126 {
127 result.Add(cycle);
128 sets.Add(new HashSet<T>(cycle));
129 }
130 }
